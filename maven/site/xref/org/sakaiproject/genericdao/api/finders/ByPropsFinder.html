74  	 * matching the supplied properties and return them in a List

76  	 * Example usage:<br/>

78  	 * <p>

80  	 * List l = findByProperties(MyClass.class, new String[] {"title"}, new Object[] {title});<br/>
81  	 * <br/>
82  	 * The list will contain all objects which have a title which matches "this title"
83  	 * 
84  	 * @param entityClass class type of the persistent object
85  	 * @param objectProperties the names of the properties of the object 
86  	 * @param values the values of the properties (can be an array of items)
87  	 * @return a List of 0 or more persistent objects
88  	 * @deprecated Use the {@link BySearchFinder#findBySearch(Class, Search)}
89  	 */
90  	@SuppressWarnings("unchecked")
91     public List findByProperties(Class entityClass, String[] objectProperties, Object[] values);

162 	 * <br/>

165 	 * @param entityClass class type of the persistent object

169 	 * use the defined constants: e.g. EQUALS, LIKE, etc...
170 	 * @param sortProperties the names of the properties of the object followed by a sort order (default is ascending),

172 	 * @return a List of 0 or more persistent objects

182 	 * <p>
183 	 * Comparisons may be used, see the constants for {@link org.sakaiproject.genericdao.api.finders.ByPropsFinder}<br/>
184 	 * Note: For the LIKE comparison you will need to specify "%" as a wildcard for string comparisons
185 	 * <p>
186 	 * Example usage:<br/>
187 	 * (Sample object of MyClass with "title" property)
188 	 * <p>
189 	 * String title = "%title";<br/>
190 	 * List l = findByProperties(MyClass.class, new String[] {"title"}, new Object[] {title}, new int[] {ByPropsFinder.LIKE}, 0, 2);<br/>
191 	 * <br/>
192 	 * The list will contain the first 2 objects (if that many exist) which have a title which ends with "title"
193 	 * <p>
194 	 * Other examples:
195 	 * List l = findByProperties(MyClass.class, new String[] {"title"}, new Object[] {title}, new int[] {ByPropsFinder.LIKE}, 5, 5);<br/>
196 	 * (<i>returns 5 objects starting with the 6th object that has a title ending with "title"</i></br>
197 	 * List l = findByProperties(MyClass.class, new String[] {"title"}, new Object[] {title}, new int[] {ByPropsFinder.LIKE}, 10, 0);<br/>
198 	 * (<i>returns all objects starting with the 11th object that has a title ending with "title"</i></br>
199 	 * 
200 	 * @param entityClass class type of the persistent object
201 	 * @param objectProperties the names of the properties of the object 
202 	 * @param values the values of the properties (can be an array of items)
203 	 * @param comparisons the comparison to make between the property and the value,
204 	 * use the defined constants: e.g. EQUALS, LIKE, etc...
205 	 * @param firstResult the index of the first result object to be retrieved (numbered from 0)
206 	 * @param maxResults the maximum number of result objects to retrieve (or <=0 for no limit)
207 	 * @return a List of 0 or more persistent objects
208     * @deprecated Use the {@link BySearchFinder#findBySearch(Class, Search)}
209 	 */

211    public List findByProperties(Class entityClass, String[] objectProperties,
