15  package org.sakaiproject.genericdao.api.mappers;
16  
17  /***
18   * Allows defining of SQL to use when executing standard queries,
19   * ? will be replaced by the values to be updated or inserted in standard SQL fashion
20   * 
21   * @author Aaron Zeckoski (azeckoski@gmail.com)
22   */
23  public interface StatementMapper extends DataMapper {
24  
25     /***
26      * Will be replaced by the table name
27      */
28     public static String TABLE_NAME = "{tableName}";
29     /***
30      * Will be replaced by a list of column names,
31      * example: ID, TITLE
32      */
33     public static String COLUMNS = "{columns}";
34     /***
35      * Will be replaced by a list of "?" corresponding to the list of columns,
36      * example: ?,?
37      */
38     public static String VALUES = "{values}";
39     /***
40      * Will be replaced by the where portion of the statement,
41      * examples: where id = ?
42      */
43     public static String WHERE = "{where}";
44     public static String UPDATE = "{update}";
45     /***
46      * Will be replaced by the select,
47      * examples: *, count(*), distinct(id)
48      */
49     public static String SELECT = "{select}";
50  
51     public String BASIC_INSERT = "INSERT INTO "+TABLE_NAME+" ("+COLUMNS+") VALUES ("+VALUES+") ";
52     public String BASIC_SELECT = "SELECT "+SELECT+" FROM "+TABLE_NAME+" "+WHERE;
53     public String BASIC_UPDATE = "UPDATE "+TABLE_NAME+" SET "+UPDATE+" "+WHERE;
54     public String BASIC_DELETE = "DELETE FROM "+TABLE_NAME+" "+WHERE;
55  
56     /***
57      * @return the template to use when generating insert statements,
58      * use the {@link #TABLE_NAME}, {@link #COLUMNS} and {@link #VALUES} constants to indicate replacements
59      * OR return null to use the default templates
60      */
61     public String getInsertTemplate();
62  
63     /***
64      * @return the template to use when generating selects,
65      * use the {@link #SELECT}, {@link #TABLE_NAME}, and {@link #WHERE} constants to indicate replacements
66      * OR return null to use the default templates
67      */
68     public String getSelectTemplate();
69  
70     /***
71      * @return the template to use when generating updates,
72      * use the {@link #TABLE_NAME}, {@link #UPDATE} and {@link #WHERE} constants to indicate replacements
73      * OR return null to use the default templates
74      */
75     public String getUpdateTemplate();
76  
77     /***
78      * @return the template to use when generating deletes,
79      * use the {@link #TABLE_NAME} and {@link #WHERE} constants to indicate replacements
80      * OR return null to use the default templates
81      */
82     public String getDeleteTemplate();
83  
84  }
