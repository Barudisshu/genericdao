23  /***
24   * Stores the names mapping from persistent entity properties to database column names,
25   * this is generated by reflection if not defined for the persistent entity
26   * 
27   * @author Aaron Zeckoski (azeckoski@gmail.com)
28   */
29  public class NamesRecord {
30  
31      private Map<String, String> propertyToColumn = new ConcurrentHashMap<String, String>();
32      private Map<String, String> columnToProperty = new ConcurrentHashMap<String, String>();
33      private Map<String, String> foreignKeyPropertyToColumn = new ConcurrentHashMap<String, String>();
34      /***
35       * Allows for special handling of the types and will convert all data going into the DB into the
36       * type indicated in this map for the column name stored here, by default this will be totally null, 
37       * will also convert the data coming out from the
38       * JDBC type to the persistent class type
39       */
40      private Map<String, Class<?>> columnToType = new ConcurrentHashMap<String, Class<?>>();
41  
42      /***
43       * Store a mapping from property to column
44       * @param property the property from a persistent object
45       * @param column the db table column which maps to the property
46       */
47      public void setNameMapping(String property, String column) {
48          setNameMapping(property, column, null);
49      }
50  
51      /***
52       * Store a mapping from property to column with a conversion type for the column
53       * @param property the property from a persistent object
54       * @param column the db table column which maps to the property
55       * @param type the conversion type to convert this to in input into the DB
56       */
57      public void setNameMapping(String property, String column, Class<?> type) {
58          if (property == null || column == null) {
59              throw new IllegalArgumentException("property and column must not be null");
60          }
61          propertyToColumn.put(property, column);
62          columnToProperty.put(column, property);
63          if (columnToProperty.size() != propertyToColumn.size()) {
64              propertyToColumn.remove(property);
65              columnToProperty.remove(column);
66              throw new IllegalArgumentException("Invalid state of mapping, there is an uneven set of properties to columns which " +
67                      "indicates that this property ("+property+") and column ("+column+") combination has overwritten an existing value");
68          }
69          if (type != null) {
70              columnToType.put(property, type);
71          }
72      }

76      /***
77       * @return the property name that represents the unique identifiers,
78       * this will be the default of "id" if not set

80      public String getIdProperty() {
81          return idProperty;

83      /***
84       * Special handling for setting the unique identifier property for a class,
85       * this is optional but recommended
86       * @param property the property from a persistent object of the unique identifier
87       * @param column the db table column which maps to the identifier property
88       */
89      public void setIdentifierProperty(String property) {
90          if (! propertyToColumn.containsKey(property)) {
91              throw new IllegalArgumentException("this property ("+property+") is not one of the mappings, " +
92                      "you must specify an id property which is an existing mapping, i.e. this cannot be " +
93              "set until after you have added the id property mapping");
94          }
95          idProperty = property;
96      }

98      /***
99       * Special handling for setting the foreign key properties for a class,
100      * this is optional but recommended
101      * @param property the property from a persistent object which matches a foreign key,
102      * normally this would be something like "thing.id" where thing is the property of the foreign key
103      * and id is the unique identifier of the object type of the thing property
104      * @param column the db table column which maps to the foreign key
105      */
106     public void setForeignKeyMapping(String property, String column) {
107         if (property == null || column == null) {
108             throw new IllegalArgumentException("property and column must not be null");
109         }
110         if (! columnToProperty.containsKey(column)) {
111             throw new IllegalArgumentException("this column ("+column+") is not one of the mappings, " +
112                     "you must specify a foreign key column which is an existing mapping, i.e. this cannot be " +
113             "set until after you have added the column mapping");
114         }
115         foreignKeyPropertyToColumn.put(property, column);
116         if (property.indexOf('.') != -1) {
117             String prefix = property.substring(0, property.indexOf('.'));
118             foreignKeyPropertyToColumn.put(prefix, column);         
119         }
120     }

123      * @param property a property from a persistent object

126     public boolean isForeignKeyProperty(String property) {
127         boolean fk = foreignKeyPropertyToColumn.containsKey(property);
128         return fk;

131     /***
132      * @param column the column name in the DB table storing data for this entity (e.g SOME_STRING)
133      * @return the property from the persistent object which maps to this db column OR null if not found
134      */
135     public String getPropertyForColumn(String column) {
136         String name = columnToProperty.get(column);
137         if (name == null && column != null) {
138             name = columnToProperty.get(column.toLowerCase());
139         }
140         if (name == null && column != null) {
141             name = columnToProperty.get(column.toUpperCase());
142         }
143         // this is bad but we need to do a case-insensitive search for the name if we still failed to find it
144         for (String key : columnToProperty.keySet()) {
145             if (key.equalsIgnoreCase(column)) {
146                 name = columnToProperty.get(key);
147                 break;
148             }
149         }
150         return name;
151     }

153     /***
154      * @param property a property in a persistent object
155      * @return the db column name which maps to this property OR null if not found
156      */
157     public String getColumnForProperty(String property) {
158         String name = foreignKeyPropertyToColumn.get(property);
159         if (name == null) {
160             name = propertyToColumn.get(property);
161         }
162         return name;
163     }

165     /***
166      * @param property a property in a persistent object
167      * @return the type which maps to this property OR null if the property name is not found or there is no conversion needed
168      */
169     public Class<?> getTypeForProperty(String property) {
170         Class<?> type = null;
171         if (property != null) {
172             String column = getColumnForProperty(property);
173             if (column != null) {
174                 type = columnToType.get(column);
175             }
176         }

181      * @param column the column name in the DB table storing data for this entity (e.g SOME_STRING)

185         Class<?> type = null;
186         if (column != null) {
187             type = columnToType.get(column);
188         }
189         return type;

192     /***
193      * Allows for special handling of the types and will convert all data going into the DB into the
194      * type indicated in this map for the property, by default this returns null for all columns
195      * indicating no conversion is needed, will also convert the data coming out from the
196      * JDBC type to the persistent class type
197      * 
198      * @param property a property in a persistent object, there must be column to property
199      * mapping for this or it will fail
200      * @param type the type which maps to this property,
201      * if null this will clear the current setting
202      * @throws IllegalArgumentException if there is no mapping for this property OR the property is null
203      */
204     public void setTypeForProperty(String property, Class<?> type) {
205         if (property == null) {
206             throw new IllegalArgumentException("property must be set");
207         }
208         String column = getColumnForProperty(property);
209         if (column == null) {
210             throw new IllegalArgumentException("No column found to match this property: " + property);
211         }
212         setTypeForColumn(column, type);
213     }

215     /***
216      * Allows for special handling of the types and will convert all data going into the DB into the
217      * type indicated in this map for the column, by default this returns null for all columns
218      * indicating no conversion is needed, will also convert the data coming out from the
219      * JDBC type to the persistent class type
220      * 
221      * @param column the column name in the DB table storing data for this entity (e.g SOME_STRING)
222      * @param type the type which maps to this property,
223      * if null this will clear the current setting
224      * @throws IllegalArgumentException if the column is null
225      */
226     public void setTypeForColumn(String column, Class<?> type) {
227         if (column == null) {
228             throw new IllegalArgumentException("column must be set");
229         }
230         if (type == null) {
231             columnToType.remove(column);
232         } else {
233             columnToType.put(column, type);
234         }
235     }

263     /***
264      * @return a list of all properties which are foreign keys,
265      * sorted in alpha order
266      */
267     public List<String> getForeignKeyPropertyNames() {
268         List<String> names = new ArrayList<String>();
269         if (foreignKeyPropertyToColumn != null) {
270             for (String fkProp : foreignKeyPropertyToColumn.keySet()) {
271                 if (fkProp.indexOf('.') == -1) {
272                     names.add( fkProp );
273                 }
274             }
275         }
276         Collections.sort(names);
277         return names;
278     }

280     /***
281      * @return a list of all column names which are foreign keys,
282      * sorted in alpha order
283      */
284     public List<String> getForeignKeyColumnNames() {
285         List<String> names = new ArrayList<String>();
286         if (foreignKeyPropertyToColumn != null) {

288                 if (fkProp.indexOf('.') == -1) {
289                     names.add( foreignKeyPropertyToColumn.get(fkProp) );
290                 }
291             }
292         }
293         Collections.sort(names);
294         return names;
