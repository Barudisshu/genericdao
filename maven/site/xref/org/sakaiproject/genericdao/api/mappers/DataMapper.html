15  package org.sakaiproject.genericdao.api.mappers;

17  import org.sakaiproject.genericdao.api.translators.DatabaseTranslator;
18  
19  /***
20   * Allows for mapping persistent object fields to a database fields and back
21   * when using the Spring JDBC generic DAO<br/>
22   * Extend it with {@link EntityColumnMapper} and {@link StatementMapper}
23   * 
24   * @author Aaron Zeckoski (azeckoski@gmail.com)
25   */
26  public interface DataMapper {
27  
28     public static String DEFAULT_ID_PROPERTY = "id";
29     public static String DEFAULT_ID_COLUMN = "ID";
30  
31     public static String DBTYPE_DB2 = DatabaseTranslator.DBTYPE_DB2;
32     public static String DBTYPE_DERBY = DatabaseTranslator.DBTYPE_DERBY;
33     public static String DBTYPE_HSQLDB = DatabaseTranslator.DBTYPE_HSQLDB;
34     public static String DBTYPE_MSSQL = DatabaseTranslator.DBTYPE_MSSQL;
35     public static String DBTYPE_MYSQL = DatabaseTranslator.DBTYPE_MYSQL;
36     public static String DBTYPE_ORACLE = DatabaseTranslator.DBTYPE_ORACLE;
37     public static String DBTYPE_POSTGRES = DatabaseTranslator.DBTYPE_POSTGRES;
38  
39     /***
40      * This string will be replaced by the value from {@link #getTableName()}
41      * for the current persistent class type
42      */
43     public static String DDL_TABLENAME = "{TABLENAME}";
44     /***
45      * This string plus a persistent class name (Class.getName()) and closed with "}" will be replaced by
46      * the name of the table for the persistent class type supplied,
47      * Example: {TABLENAME:org.domain.MyClass} would be replaced by the value 
48      * returned by table name for the persistent type MyClass
49      */
50     public static String DDL_TABLENAME_TYPE_PREFIX = "{TABLENAME:";
51     /***
52      * This string will be replaced by the column name of the unique identifier
53      * for the current persistent class type
54      */
55     public static String DDL_ID_COLUMN = "{ID}";
56     /***
57      * This string plus a persistent class name (Class.getName()) and closed with "}" will be replaced by
58      * the unique identifier for the persistent class type supplied,
59      * Example: {ID:org.domain.MyClass} would be replaced by the 
60      * column name of the unique identifier for the persistent type MyClass
61      */
62     public static String DDL_ID_TYPE_PREFIX = "{ID:";
63     /***
64      * (Oracle ONLY) a sequence name will be generated and inserted
65      * based on the tablename for use in generating IDs,
66      * if you want to specify your own sequence name then you will lose
67      * the ability to have the ID inserted into newly created objects
68      */
69     public static String DDL_ID_SEQNAME = "{IDSEQNAME}";
70     /***
71      * This string plus a property name and closed with "}" will be replaced by
72      * the column name which maps to that property for the current persistent class type,
73      * Example: {COLUMNNAME:title} would be replaced by the column name which maps to the title property
74      */
75     public static String DDL_COLUMN_PREFIX = "{COLUMNNAME:";
76  
77     /***
78      * @return the class type of the persistent objects handled by this data mapper
79      */
80     public Class<?> getPersistentType();
81  
82     /***
83      * @return the name of the table where these persistent objects are stored
84      * OR return null to have generic dao generate the name from the classname
85      */
86     public String getTableName();
87  
88     /***
89      * Specify the value which is the unique id for this persistent object,
90      * normally the values on this property should be assigned by the database (e.g. autoincrement)
91      * 
92      * @return the name of the identifier property from the persistent object
93      * OR return null to use the default "id"
94      */
95     public String getIdPropertyName();
96  
97     /***
98      * Defines the mapping between persistent entity properties and database column names,
99      * NOTE: auto generate will surely fail if you did not use the standard UPPERCASE and underscore
100     * to create your column names in your DDL (e.g. "myThing" -> "MY_THING")
101     * 
102     * @return a {@link NamesRecord} 
103     * OR return null to have generic dao generate the mapping using reflection
104     */
105    public NamesRecord getPropertyToColumnNamesMapping();
106    
107    /***
108     * Called when the JDBC generic dao beans are first starting up,
109     * if the first line fails then the rest of the file is ignored,
110     * this means you should normally include the create table statement first<br/>
111     * The first non-comment ('--') line will be run, and if successful, 
112     * all other non-comment lines will be run. SQL statements may be on 
113     * multiple lines but must have ';' terminators.<br/>
114     * The following keys will be replaced automatically:<br/>
115     * {TABLENAME} - the value returned by {@link #getTableName()}<br/>
116     * {ID} - the column name of the unique identifier<br/>
117     * {TABLENAME:org.domain.MyClass} - the value returned by {@link #getTableName()} for the persistent type MyClass<br/>
118     * {ID:org.domain.MyClass} - the column name of the unique identifier for the persistent type MyClass<br/>
119     * {IDSEQNAME} - (Oracle) a sequence name will be generated and inserted
120     * based on the tablename for use in generating IDs,
121     * if you want to specify your own sequence name then you will lose
122     * the ability to have the ID inserted into newly created objects<br/>
123     * 
124     * @param databaseTypeConstant one of the database type constants from, e.g. {@link #DBTYPE_MYSQL},
125     * this indicates which database your DDL should work for,
126     * you can throw an exception here to indicate you do not support the given database type
127     * @return the database definition language string which will create the table for this persistent object to be stored in
128     */
129    public String generateDDL(String databaseTypeConstant);
130 
131 }
