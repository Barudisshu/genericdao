19  /***
20   * A simple bean which defines a restriction in a search, this is like saying:
21   * where userId = '123'; OR where userId like '%aaronz%';<br/>
22   * Example usage:<br/>
23   * <code>Restriction rteq = new Restriction("title", curTitle); // restrict search to title equals value of curTitle</code><br/>
24   * <code>Restriction rtne = new Restriction("title", curTitle, Restriction.NOT_EQUALS); // restrict search to title not equals value of curTitle</code><br/>
25   *
26   * @author Aaron Zeckoski (azeckoski@gmail.com)
27   */

74      public String getStringValue() {
75          String s = "";
76          if (value != null) {

86      /***

88       */
89      public boolean getBooleanValue() {
90          Object o = getSingleValue();
91          if (o != null) {
92              if (o instanceof Boolean) {
93                  return (Boolean) o;
94              } else if (o instanceof String) {
95                  return new Boolean((String)o).booleanValue();
96              }
97          }
98          return false;
99      }

101     /***
102      * @return the value if it is not an array or collection,
103      * if it is then return the first object in the array or collection,
104      * otherwise return null (including when value is null)
105      */
106     @SuppressWarnings("unchecked")
107     public Object getSingleValue() {
108         Object result = null;
109         if (value != null) {
110             if (value.getClass().isArray()) {
111                 if (((Object[])value).length > 0) {
112                     result = ((Object[])value)[0];
113                 }
114             } else if (value.getClass().isAssignableFrom(Collection.class)) {
115                 if (! ((Collection)value).isEmpty()) {
116                     result = ((Collection)value).iterator().next();
117                 }
118             } else {
119                 result = value;
120             }
121         }
122         return result;
123     }
124 
125     /***
126      * @return the value as an array if it is an array or a collection,
127      * if it is a single value then return null (also when value is null)
128      */
129     @SuppressWarnings("unchecked")
130     public Object[] getArrayValue() {
131         Object[] result = null;
132         if (value != null) {
133             if (value.getClass().isArray()) {
134                 result = (Object[])value;
135             } else if (value.getClass().isAssignableFrom(Collection.class)) {
136                 result = ((Collection)value).toArray();
137             }
138         }      
139         return result;
140     }
