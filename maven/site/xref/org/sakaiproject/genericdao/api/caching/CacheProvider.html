15  package org.sakaiproject.genericdao.api.caching;
16  
17  
18  /***
19   * Allows the developer to specify their own cache mechanism which will be used for all
20   * generic DAO method calls, write methods will automatically cause the cache to expire
21   * any related items<br/>
22   * Note that your cache must be able to store null values 
23   * (these are used to cache searches which come up with no results and are critical since these
24   * are often the most expensive searches and can cause full table scans)
25   * 
26   * @author Aaron Zeckoski (azeckoski@gmail.com)
27   */
28  public interface CacheProvider {
29  
30     /***
31      * Create a cache by the given name (or if the cache already exists then reset it),
32      * the cache will be used to cache one type of persistent objects,
33      * the cache only needs to exist in your implementation as it will be accessed using the given name
34      * @param cacheName a string which uniquely identifies this cache
35      * @throws IllegalArgumentException if the cache name is invalid
36      * @throws IllegalStateException if the cache cannot be created
37      */
38     public void createCache(String cacheName);
39  
40     /***
41      * Puts an object in the specified cache
42      * @param cacheName a string which uniquely identifies this cache
43      * @param key the key for a persistent object (this will be the persistent id)
44      * @param value a persistent object (this can be a null to cache a miss)
45      * @throws IllegalArgumentException if the cache name is invalid or cacheName or key is null
46      */
47     public void put(String cacheName, String key, Object value);
48  
49     /***
50      * Gets an object from the cache if it can be found (maybe be a null),
51      * use the exists check to see if the object is in the cache before retrieving
52      * @param cacheName a string which uniquely identifies this cache
53      * @param key the key for a persistent object (this will be the persistent id)
54      * @return the cached persistent object (may be null)
55      * @throws IllegalArgumentException if the cache name is invalid or any arguments are null
56      * @throws CacheKeyNotFoundException if this key does not exist in the cache
57      */
58     public Object get(String cacheName, String key);
59  
60     /***
61      * Removes an object from the cache if it exists or does nothing
62      * @param cacheName a string which uniquely identifies this cache
63      * @param key the key for a persistent object (this will be the persistent id)
64      * @return true if the object was removed or false if it could not be found in the cache
65      * @throws IllegalArgumentException if the cache name is invalid or any arguments are null
66      */
67     public boolean remove(String cacheName, String key);
68  
69     /***
70      * Check if a key exists in the cache and return true if it does
71      * @param cacheName a string which uniquely identifies this cache
72      * @param key the key for a persistent object (this will be the persistent id)
73      * @return true if the object was removed or false if it could not be found in the cache
74      * @throws IllegalArgumentException if the cache name is invalid or any arguments are null
75      */
76     public boolean exists(String cacheName, String key);
77  
78     /***
79      * Clear out all cached items from this cache
80      * @param cacheName a string which uniquely identifies this cache
81      * @throws IllegalArgumentException if the cache name is invalid
82      */
83     public void clear(String cacheName);
84  
85  }
