15  package org.sakaiproject.genericdao.springutil;
16  
17  import java.io.PrintWriter;
18  import java.sql.Connection;
19  import java.sql.SQLException;
20  
21  import javax.sql.DataSource;
22  
23  import org.springframework.jdbc.datasource.SmartDataSource;
24  
25  
26  /***
27   * This allows us to trick spring into not handling our dataSources for us,
28   * it keeps spring from automatically closing every non-spring DataSource connection
29   * after each method is executed
30   * 
31   * @author Aaron Zeckoski (azeckoski @ gmail.com)
32   */
33  public class SmartDataSourceWrapper implements DataSource, SmartDataSource {
34  
35      private final DataSource dataSource;
36  
37      /***
38       * Create a new data-source wrapper,
39       * all connections are set to auto-commit false by default
40       * 
41       * @param dataSource any non-spring {@link DataSource}
42       */
43      public SmartDataSourceWrapper(DataSource dataSource) {
44          this.dataSource = dataSource;
45      }
46  
47      private boolean autoCommitConnection = false;
48      /***
49       * @param autoCommitConnection the auto-commit for all connections coming out is set to this
50       */
51      public void setAutoCommitConnection(boolean autoCommitConnection) {
52          this.autoCommitConnection = autoCommitConnection;
53      }
54      /***
55       * Fixes up the auto-commit to be equal to the setting for this connection if possible,
56       * will not cause an exception though if it fails to do it
57       * @param connection the connection
58       * @return the same connection that was input
59       */
60      private Connection fixAutoCommit(Connection connection) {
61          try {
62              if (connection.getAutoCommit() != autoCommitConnection) {
63                  connection.setAutoCommit(autoCommitConnection);
64              }
65          } catch (SQLException e) {
66              // do nothing
67          }
68          return connection;
69      }
70  
71      // force spring to not close out this connection for us
72      public boolean shouldClose(Connection con) {
73          return false;
74      }
75  
76  
77      public Connection getConnection() throws SQLException {
78          return fixAutoCommit( dataSource.getConnection() );
79      }
80      public Connection getConnection(String username, String password) throws SQLException {
81          return fixAutoCommit( dataSource.getConnection(username, password) );
82      }
83      public PrintWriter getLogWriter() throws SQLException {
84          return dataSource.getLogWriter();
85      }
86      public int getLoginTimeout() throws SQLException {
87          return dataSource.getLoginTimeout();
88      }
89      public void setLogWriter(PrintWriter out) throws SQLException {
90          dataSource.setLogWriter(out);
91      }
92      public void setLoginTimeout(int seconds) throws SQLException {
93          dataSource.setLoginTimeout(seconds);
94      }
95      // Java 6 compatible
96      public boolean isWrapperFor(Class<?> iface) throws SQLException {
97          return false;
98      }
99      public <T> T unwrap(Class<T> iface) throws SQLException {
100         throw new SQLException("Not a Wrapper for " + iface);
101     }
102 
103 }
