29  /***
30   * This sets the {@link ClassLoader} for the {@link BeanNameAutoProxyCreator} to the current
31   * one that this class exists in<br/>
32   * Compatible with Spring 1.2.8+
33   * 
34   * @author Aaron Zeckoski (azeckoski@gmail.com)
35   */
36  public class CurrentClassLoaderBeanNameAutoProxyCreator extends BeanNameAutoProxyCreator {
37  	private static final long serialVersionUID = 1L;
38  
39  	protected ClassLoader myClassLoader = CurrentClassLoaderBeanNameAutoProxyCreator.class.getClassLoader();
40  
41  	/**** only works with Spring 2.0.x
42     @Override
43     public void setBeanClassLoader(ClassLoader classLoader) {
44        // this is basically ignoring the input classLoader and setting it to the one
45        // which this class is currently part of
46        super.setBeanClassLoader(myClassLoader);

50  	// below needed to make this work with spring 1.2.8
51  	private boolean freezeProxy = false;
52  	@Override
53  	public void setFrozen(boolean frozen) {
54  		this.freezeProxy = frozen;
55  	}
56  	@Override
57  	public boolean isFrozen() {
58  		return this.freezeProxy;
59  	}

61  	protected String[] interceptorNames = new String[0];
62  	@Override
63  	public void setInterceptorNames(String[] interceptorNames) {
64  		this.interceptorNames = interceptorNames;
65  	}
66  
67  	private AdvisorAdapterRegistry advisorAdapterRegistry = GlobalAdvisorAdapterRegistry.getInstance();
68  	@Override
69  	public void setAdvisorAdapterRegistry(AdvisorAdapterRegistry advisorAdapterRegistry) {
70  		this.advisorAdapterRegistry = advisorAdapterRegistry;
71  	}

73  	private boolean applyCommonInterceptorsFirst = true;
74  	@Override
75  	public void setApplyCommonInterceptorsFirst(boolean applyCommonInterceptorsFirst) {
76  		this.applyCommonInterceptorsFirst = applyCommonInterceptorsFirst;
77  	}
78  
79  	@SuppressWarnings("unchecked")
80  	protected boolean shouldProxyTargetClass(Class beanClass, String beanName) {
81  		// note that we are ignoring the ConfigurableListableBeanFactory check
82  		return isProxyTargetClass();
83  	}

85  	private Advisor[] resolveInterceptorNames() {
86  		Advisor[] advisors = new Advisor[this.interceptorNames.length];
87  		for (int i = 0; i < this.interceptorNames.length; i++) {
88  			Object next = getBeanFactory().getBean(this.interceptorNames[i]);
89  			advisors[i] = this.advisorAdapterRegistry.wrap(next);
90  		}
91  		return advisors;
92  	}
93  
94  	@SuppressWarnings("unchecked")
95  	protected Advisor[] buildAdvisors(String beanName, Object[] specificInterceptors) {
96  		// Handle prototypes correctly...
97  		Advisor[] commonInterceptors = resolveInterceptorNames();
98  
99  		List allInterceptors = new ArrayList();
100 		if (specificInterceptors != null) {
101 			allInterceptors.addAll(Arrays.asList(specificInterceptors));
102 			if (commonInterceptors != null) {
103 				if (this.applyCommonInterceptorsFirst) {
104 					allInterceptors.addAll(0, Arrays.asList(commonInterceptors));
105 				}
106 				else {
107 					allInterceptors.addAll(Arrays.asList(commonInterceptors));
108 				}
109 			}
110 		}
111 		//      if (logger.isDebugEnabled()) {
112 			//         int nrOfCommonInterceptors = (commonInterceptors != null ? commonInterceptors.length : 0);
113 		//         int nrOfSpecificInterceptors = (specificInterceptors != null ? specificInterceptors.length : 0);
114 		//         logger.debug("Creating implicit proxy for bean '" + beanName + "' with " + nrOfCommonInterceptors +
115 		//               " common interceptors and " + nrOfSpecificInterceptors + " specific interceptors");
116 		//      }
117 
118 		Advisor[] advisors = new Advisor[allInterceptors.size()];
119 		for (int i = 0; i < allInterceptors.size(); i++) {
120 			advisors[i] = this.advisorAdapterRegistry.wrap(allInterceptors.get(i));
121 		}
122 		return advisors;
123 	}
124 
125 	@SuppressWarnings("unchecked")
126 	@Override
127 	protected Object createProxy(
128 			Class beanClass, String beanName, Object[] specificInterceptors, TargetSource targetSource) {
129 
130 		ProxyFactory proxyFactory = new ProxyFactory();
131 		// Copy our properties (proxyTargetClass etc) inherited from ProxyConfig.
132 		proxyFactory.copyFrom(this);
133 
134 		if (!shouldProxyTargetClass(beanClass, beanName)) {
135 			// Must allow for introductions; can't just set interfaces to
136 			// the target's interfaces only.
137 			Class[] targetInterfaces = ClassUtils.getAllInterfacesForClass(beanClass);
138 			for (int i = 0; i < targetInterfaces.length; i++) {
139 				proxyFactory.addInterface(targetInterfaces[i]);
140 			}
141 		}
142 
143 		Advisor[] advisors = buildAdvisors(beanName, specificInterceptors);
144 		for (int i = 0; i < advisors.length; i++) {
145 			proxyFactory.addAdvisor(advisors[i]);
146 		}
147 
148 		proxyFactory.setTargetSource(targetSource);
149 		customizeProxyFactory(proxyFactory);
150 
151 		proxyFactory.setFrozen(this.freezeProxy);
152 		return proxyFactory.getProxy(myClassLoader);
153 	}
