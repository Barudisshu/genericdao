17  /***
18   * This is needed to keep the {@link ClassLoader} from getting overwritten after we set it,
19   * the {@link ClassLoader} can only be set one time using the setter method<br/>
20   * 
21   * @author Aaron Zeckoski (azeckoski@gmail.com)
22   */
23  public class SetOnceClassLoaderTxProxyFactoryBean extends CurrentClassLoaderTxProxyFactoryBean {
24  	private static final long serialVersionUID = 1L;
25  
26  	boolean alreadySet = false;
27  	public void setProxyClassLoader(ClassLoader classLoader) {
28  		if (classLoader != null) {
29  			if (! alreadySet) {
30  				this.myClassLoader = classLoader;
31  				alreadySet = true;
32  			}
33  		}
34  	}
35  
36  }
