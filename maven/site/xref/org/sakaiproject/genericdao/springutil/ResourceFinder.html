23  import org.springframework.core.io.ClassPathResource;
24  import org.springframework.core.io.Resource;

26  /***
27   * Allows for easy resolution of resources in the current {@link ClassLoader},
28   * (which may not be the {@link ClassLoader} being used by Spring)<br/>
29   * Primarily useful when you need to locate resources while running in a Sakai component
30   * (like HBM files or SQL files)<br/>
31   * In Sakai, this allows us to find resources in our pack since the Sakai context {@link ClassLoader} is wrong,
32   * too bad it is not correct, that would be cool, but it is wrong and it is not cool<br/>
33   * Takes a list of paths to resources and turns them into legitimate resources<br/>
34   * <xmp><bean class="org.sakaiproject.genericdao.springutil.ResourceFinder" factory-method="getResources">
35         <constructor-arg>
36             <list>
37                 <value>org/sakaiproject/blogwow/dao/hbm/BlogWowBlog.hbm.xml</value>
38                 <value>org/sakaiproject/blogwow/dao/hbm/BlogWowEntry.hbm.xml</value>
39                 <value>org/sakaiproject/blogwow/dao/hbm/BlogWowComment.hbm.xml</value>
40             </list>
41         </constructor-arg>
42     </bean></xmp>
43   * 
44   * @author Aaron Zeckoski (azeckoski@gmail.com)
45   */
46  public class ResourceFinder {
47  
48     private static List<Resource> makeResources(List<String> paths) {
49        List<Resource> rs = new ArrayList<Resource>();
50        if (paths != null && !paths.isEmpty()) {
51           ClassLoader cl = ResourceFinder.class.getClassLoader();
52           for (String path : paths) {
53              Resource r = new ClassPathResource(path, cl);
54              if (r.exists()) {
55                 rs.add(r);
56              }
57           }
58        }
59        return rs;
60     }
61  
62     /***
63      * Resolves a list of paths into resources within the current classloader
64      * @param paths a list of paths to resources (org/sakaiproject/mystuff/Thing.xml)
65      * @return an array of Spring Resource objects
66      */
67     public static Resource[] getResources(List<String> paths) {
68        List<Resource> l = makeResources(paths);
69        return l.toArray(new Resource[l.size()]);
70     }
71  
72     /***
73      * Resolves a list of paths into resources within the current classloader
74      * @param paths a list of paths to resources (org/sakaiproject/mystuff/Thing.xml)
75      * @return an array of File objects
76      */
77     public static File[] getFiles(List<String> paths) {
78        List<Resource> rs = makeResources(paths);
79        File[] files = new File[rs.size()];
80        for (int i = 0; i < rs.size(); i++) {
81           Resource r = rs.get(i);
82           try {
83              files[i] = r.getFile();
84           } catch (IOException e) {
85              throw new RuntimeException("Failed to get file for: " + r.getFilename(), e);
86           }
87        }
88        return files;
89     }
90  
91     /***
92      * Resolves a list of paths into resources within the current classloader
93      * @param paths a list of paths to resources (org/sakaiproject/mystuff/Thing.xml)
94      * @return an array of InputStreams
95      */
96     public static InputStream[] getInputStreams(List<String> paths) {
97        List<Resource> rs = makeResources(paths);
98        InputStream[] streams = new InputStream[rs.size()];
99        for (int i = 0; i < rs.size(); i++) {
100          Resource r = rs.get(i);
101          try {
102             streams[i] = r.getInputStream();
103          } catch (IOException e) {
104             throw new RuntimeException("Failed to get inputstream for: " + r.getFilename(), e);
105          }
106       }
107       return streams;
108    }
