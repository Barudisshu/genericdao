28  /***
29   * This is a connection wrapper that allows a close hook to be added to the connection
30   * 
31   * @author Aaron Zeckoski (azeckoski @ gmail.com)
32   */
33  public class CloseHookConnectionWrapper implements Connection {
34  
35      private final Connection wrappedConnection;
36      private final Runnable closeHook;
37  
38      /***
39       * Create a connection wrapper which allows a close hook to be added
40       * @param connection the connection to wrap
41       * @param closeHook execute this hook when the connection closes
42       */
43      public CloseHookConnectionWrapper(Connection connection, Runnable closeHook) {
44          this.wrappedConnection = connection;
45          this.closeHook = closeHook;
46      }
47  
48      public void close() throws SQLException {
49          try {
50              wrappedConnection.close();
51          } finally {
52              // execute the runnable
53              if (closeHook != null) {
54                  try {
55                      closeHook.run();
56                  } catch (Exception e) {
57                      System.out.println("Unable to execute the runnable on connection close: " + e.getMessage());
58                  }
59              }
60          }
61      }
62  
63      @Override
64      public String toString() {
65          return "conn="+wrappedConnection+":"+super.toString();
66      }
