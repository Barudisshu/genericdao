33      /***
34       * Make a partial SQL string for a specific entity type using the associated {@link NamesRecord}
35       * and a {@link Search} object
36       * @param namesRecord a record of the mapping of property to columns for an entity type
37       * @param search a search object
38       * @return an object with the partial where SQL query and the params (args)
39       */
40      public static QueryData makeSQLfromSearch(NamesRecord namesRecord, Search search) {
41          // Checks to see if the required params are set and throws exception if not
42          if (search == null) {
43              throw new IllegalArgumentException("search cannot be null");
44          }
45  
46          QueryData sd = new QueryData();
47          //NamesRecord nr = getNamesRecord(type);
48          StringBuilder sql = new StringBuilder();
49  
50          // Only add in restrictions if there are some to add
51          if (search.getRestrictions() != null && search.getRestrictions().length > 0) {
52              // set up the conjunction
53              String junction = " and ";
54              if (! search.conjunction) {
55                  // set to use disjunction
56                  junction = " or ";
57              }
58  
59              // put in the restrictions
60              for (int i = 0; i < search.getRestrictions().length; i++) {
61                  if (i > 0) {
62                      sql.append(junction);
63                  }
64                  String property = search.getRestrictions()[i].property;
65                  Object value = search.getRestrictions()[i].value;
66                  if (property == null || value == null) {
67                      throw new IllegalArgumentException("restrictions property and value cannot be null or empty");            
68                  }
69  
70                  String column = namesRecord.getColumnForProperty(property);
71                  // we cannot recover from the case where the property is invalid
72                  if (column == null) {
73                  	throw new IllegalArgumentException("Cannot find a column which matches the property ("+property+") for search: " + search);
74                  }
75                  if (value.getClass().isArray()) {
76                      // special handling for "in" type comparisons
77                      Object[] objectArray = (Object[]) value;
78                      if (objectArray.length == 1) {
79                          value = objectArray[0];
80                      } else if (objectArray.length > 1) {
81                          sql.append(column);
82                          if (Restriction.NOT_EQUALS == search.getRestrictions()[i].comparison) {
83                              sql.append(" not in ");
84                          } else {
85                              sql.append(" in ");
86                          }
87                          // need to create this kind of thing: (?,?,?) where there are 3 things in the array
88                          sql.append("(");
89                          for (int j = 0; j < objectArray.length; j++) {
90                              if (j > 0) { sql.append(','); }
91                              sql.append('?');
92                          }
93                          sql.append(") ");
94                          // now we need to add the array of values (and convert possibly all of them)
95                          for (int j = 0; j < objectArray.length; j++) {
96                              sd.addArg( convertColumn(namesRecord, column, objectArray[j]) );
97                          }
98                          //sd.addArrayArg(objectArray);
99                      } else {
100                         // do nothing for now, this is slightly invalid but not worth dying over
101                     }
102                 }
103 
104                 if (! value.getClass().isArray()) {
105                     int comparisonConstant = search.getRestrictions()[i].comparison;
106                     // convert if needed
107                     value = convertColumn(namesRecord, column, value);
108                     sql.append( makeComparisonSQL(sd.args, namesRecord.getColumnForProperty(property), comparisonConstant, value) );
109                 }
110             }
111         }
112 
113         // handle the sorting (sort param can be null for no sort)
114         if (search.getOrders() != null) {
115             StringBuilder orderSQL = new StringBuilder();
116             if (search.getOrders().length > 0) {
117                 orderSQL.append(" order by ");
118                 for (int i = 0; i < search.getOrders().length; i++) {
119                     if (i > 0) {
120                         orderSQL.append(", ");
121                     }
122                     String property = search.getOrders()[i].property;
123                     orderSQL.append(namesRecord.getColumnForProperty(property));
124                     if (search.getOrders()[i].ascending) {
125                         orderSQL.append(" asc ");
126                     } else {
127                         orderSQL.append(" desc ");
128                     }
129                 }
130             }
131             sd.orderSQL = orderSQL.toString();
132         }
133 
134         sd.whereSQL = sql.toString();
135         return sd;
136     }

138     /***
139      * Create comparison SQL but converts the value object to a string
140      * 
141      * @param column the name of a database column
142      * @param comparisonConstant the comparison constant (e.g. EQUALS)
143      * @param value the value to compare the property to
144      * @return a string representing the SQL snippet (e.g. propA = ?)
145      */
146     public static String makeComparisonSQL(String column, int comparisonConstant, Object value) {
147         String sval = null;
148         if (comparisonConstant != Restriction.NOT_NULL 
149                 && comparisonConstant != Restriction.NULL) {
150             if (value.getClass().isAssignableFrom(Boolean.class) 
151                     || value.getClass().isAssignableFrom(Number.class)) {
152                 // special handling for boolean and numbers
153                 sval = value.toString();
154             } else {
155                 sval = "'" + value.toString() + "'";
156             }
157         }
158         return buildComparisonSQL(column, comparisonConstant, sval);
159     }

161     /***
162      * Create comparison SQL but places the values into the params list
163      * 
164      * @param params a set of params to add this value to
165      * @param column the name of a database column
166      * @param comparisonConstant the comparison constant (e.g. EQUALS)
167      * @param value the value to compare the property to
168      * @return a string representing the SQL snippet (e.g. propA = ?)
169      */
170     public static String makeComparisonSQL(List<Object> params, String column, int comparisonConstant, Object value) {
171         if (comparisonConstant != Restriction.NOT_NULL 
172                 && comparisonConstant != Restriction.NULL) {
173             params.add(value);
174         }
175         return buildComparisonSQL(column, comparisonConstant, "?");
176     }

178     /***
179      * Convenient way to translate constants into SQL,
180      * sVal can be any string value including ?
181      */
182     protected static String buildComparisonSQL(String column, int comparisonConstant, String sval) {
183         switch (comparisonConstant) {
184         case Restriction.EQUALS:      return column + " = " + sval;
185         case Restriction.GREATER:     return column + " > " + sval;
186         case Restriction.LESS:        return column + " < " + sval;
187         case Restriction.LIKE:        return column + " like " + sval;
188         case Restriction.NOT_EQUALS:  return column + " <> " + sval;
189         case Restriction.NOT_NULL:    return column + " is not null";
190         case Restriction.NULL:        return column + " is null";
191         default: throw new IllegalArgumentException("Invalid comparison constant: " + comparisonConstant);
192         }
193     }

195     /***
196      * This helper method will convert the incoming data if it needs to be
197      * converted for the given column, otherwise it will do nothing to the value
198      * @param namesRecord the names record for the persistent class
199      * @param column the name of the column this value is associated with
200      * @param value the value to convert
201      * @return the converted value or the original value if no conversion needed
202      */
203     public static Object convertColumn(NamesRecord namesRecord, String column, Object value) {
204         if (namesRecord != null && column != null) {
205             try {
206                 Class<?> convertType = namesRecord.getTypeForColumn(column);
207                 if (convertType != null) {
208                     value = ConversionUtils.getInstance().convert(value, convertType);
209                 }
210             } catch (Exception e) {
211                 // nothing to do but continue on
212             }
213         }
214         return value;
215     }

219     /***
220      * Special class for passing SQL query data for running queries with,
221      * has methods to get the where statement and the set of params (args)
222      * 
223      * @author Aaron Zeckoski (azeckoski@gmail.com)
224      */
225     public static class QueryData {
226         public String whereSQL = "";
227         public String orderSQL = "";
228         public List<Object> args = new ArrayList<Object>();
229         /***
230          * Add an argument value to the arg array
231          * @param value an object
232          */
233         public void addArg(Object value) {
234             args.add(value);
235         }
236         /*** 
237          * Add an array of argument values to the arg array
238          * @param values an array of objects
239          */
240         public void addArrayArg(Object[] values) {
241             for (int i = 0; i < values.length; i++) {
242                 addArg(values[i]);
243             }
244         }
245         /***
246          * Get the SQL statement to place after the tables in the SQL statement,
247          * typically this is something like "where name like '%aaronz%' order by name"
248          * @return the string to place after the table list, will not be null
249          */
250         public String getAfterTableSQL() {
251             String sql = "";
252             if (whereSQL != null &&  !("".equals(whereSQL))) {
253                 sql = " where " + whereSQL;
254             }
255             if (orderSQL != null &&  !("".equals(orderSQL))) {
256                 sql += orderSQL;
257             }
258             return sql;
259         }
260         /***
261          * @return the array of arguments which should replace the ? is the sql string
262          */
263         public Object[] getArgs() {
264             return args.toArray();
265         }
266     }
