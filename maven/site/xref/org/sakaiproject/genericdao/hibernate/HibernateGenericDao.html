58     /***
59      * Set the list of persistent classes that this DAO will handle<br/>
60      * This no longer requires this to be set before the class initializes
61      * Do this using Spring like so:
62      * <xmp>
63        <property name="persistentClasses">
64           <list>
65              <value>org.sakaiproject.*yourappname*.model.*YourObject*</value>
66           </list>
67        </property>
68      * </xmp>
69      * @param classes a list of Strings representing the
70      * the fully qualified classpath and classname of persistent objects
71      */
72     public void setPersistentClasses(List<String> classes) {
73        if (classes.size() == 0) {
74           throw new IllegalArgumentException("persistent class list must have at least one item");
75        }
76        List<Class<?>> classList = new ArrayList<Class<?>>();
77        for (Iterator<String> i = classes.iterator(); i.hasNext();) {
78           String className = i.next();
79           Class<?> c = ClassLoaderUtils.getClassFromString(className);
80           if (c == null) {
81              throw new IllegalArgumentException(
82                    "Invalid class type in list of persistent classes: " + className);
83           }
84           classList.add(c);
85        }
86        this.classes = classList;
87     }

90     /***
91      * Take a list of persistent hibernate objects and return non-persistent clones
92      * @param items a list of persistent objects
93      * @return a list of non-persistent cloned objects
94      */
95     protected <T> List<T> cloneList(List<T> items) {
96        if (items.isEmpty()) { return items; }
97        List<T> clonedItems = new ArrayList<T>();
98        for (T item : items) {
99           // create a clone of each hibernate object and pass back the clone only,
100          // cloning only the bottom level of data
101          clonedItems.add( ReflectUtils.getInstance().clone(item, 0, null) );
102       }
103       return clonedItems;
104    }

112     * it is efficient in that it does not return all objects and only does the count

135    /***
136     * Builds count HQL from an HQL query
137     * 
138     * @param hqlQuery an hql query
139     * @return the HQL to do a count
140     */
141    private String buildCountHQL(String hqlQuery) {
142       if (hqlQuery == null) {
143          throw new IllegalArgumentException("hqlQuery cannot be null");
144       }
145       hqlQuery = hqlQuery.trim();
146 
147       // clean HQL up in case it looks like this: select blah from Blah
148       int fromLoc = hqlQuery.indexOf(START_QUERY);
149       if (fromLoc == -1) {
150          throw new IllegalArgumentException("HQL appears to be invalid: "
151                + "query does not start with the string: " + START_QUERY);
152       }
153 
154       // get rid of the sort by since this will cause the count query to fail
155       int sortLoc = hqlQuery.indexOf("sort by");
156       if (sortLoc == -1) { sortLoc = hqlQuery.length(); }
157 
158 
159       String newHqlQuery = hqlQuery.substring(fromLoc, sortLoc);
160       if (fromLoc == 0) {
161          newHqlQuery = COUNTALL_QUERY + newHqlQuery;
162       } else {
163          // make sure we keep the "blah" part and just put count() around it: select count(blah) from Blah
164          String oldPrefix = hqlQuery.substring(0, fromLoc);
165          String[] parts = oldPrefix.split(" ");
166          if (parts.length > 1) {
167             String prefix = "select count(" + parts[1] + ") ";
168             newHqlQuery = prefix + newHqlQuery;
169          }
170       }
171       return newHqlQuery;
172    }

175    /***
176     * Generates the HQL snippet needed to represent this property/comparison/value triple,
177     * will turn the value into a string, use {@link #makeComparisonHQL(Map, String, int, Object)}
178     * if you want to deliver the object directly
179     * 
180     * @param property the name of the entity property
181     * @param comparisonConstant the comparison constant (e.g. EQUALS)
182     * @param value the value to compare the property to
183     * @return a string representing the HQL snippet (e.g. propA = 'apple')
184     */
185    protected String makeComparisonHQL(String property, int comparisonConstant, Object value) {
186       String sval = null;
187       if (comparisonConstant != Restriction.NOT_NULL && comparisonConstant != Restriction.NULL) {
188          if (value.getClass().isAssignableFrom(Boolean.class) 
189                || value.getClass().isAssignableFrom(Number.class)) {
190             // special handling for boolean and numbers
191             sval = value.toString();
192          } else {
193             sval = "'" + value.toString() + "'";
194          }
195       }
196       return buildComparisonHQL(property, comparisonConstant, sval);
197    }
198 
199    /***
200     * @param params a set of params to add this value to
201     * @param property the name of the entity property
202     * @param comparisonConstant the comparison constant (e.g. EQUALS)
203     * @param value the value to compare the property to
204     * @return a string representing the HQL snippet (e.g. propA = :propA)
205     */
206    protected String makeComparisonHQL(Map<String, Object> params, String property, int comparisonConstant, Object value) {
207       if (comparisonConstant != Restriction.NOT_NULL && comparisonConstant != Restriction.NULL) {
208          params.put(property, value);
209       }
210       return buildComparisonHQL(property, comparisonConstant, ":" + property);
211    }

213    /***
214     * @param property
215     * @param comparisonConstant
216     * @param sval
217     * @return
218     */
219    private String buildComparisonHQL(String property, int comparisonConstant, String sval) {
220       switch (comparisonConstant) {
221       case Restriction.EQUALS:      return property + " = " + sval;
222       case Restriction.GREATER:     return property + " > " + sval;
223       case Restriction.LESS:        return property + " < " + sval;
224       case Restriction.LIKE:        return property + " like " + sval;
225       case Restriction.NOT_EQUALS:  return property + " <> " + sval;
226       case Restriction.NOT_NULL:    return property + " is not null";
227       case Restriction.NULL:        return property + " is null";
228       default: throw new IllegalArgumentException("Invalid comparison constant: " + comparisonConstant);
229       }
230    }

233    /***
234     * Provides an easy way to execute an HQL query with ? params
235     * 
236     * @param hql a hibernate query language query
237     * @param params an array of values (you should have "?" in your HQL whereever the values will be inserted)
238     * @param start the entry number to start on (based on current sort rules), first entry is 0
239     * @param limit the maximum number of entries to return, 0 returns as many entries as possible
240     * @return a list of whatever you requested in the HQL
241     */
242    protected List<?> executeHqlQuery(String hql, Object[] params, int start, int limit) {
243       Query query = getSession().createQuery(hql);
244       query.setFirstResult(start);
245       if (limit > 0) {
246          query.setMaxResults(limit);
247       }
248       for (int i = 0; i < params.length; i++) {
249          query.setParameter(i, params[i]);
250       }
251       return query.list();
252    }

254    /***
255     * Provides an easy way to execute an HQL query with named parameters
256     * 
257     * @param hql a hibernate query language query
258     * @param params the map of named parameters
259     * @param start the entry number to start on (based on current sort rules), first entry is 0
260     * @param limit the maximum number of entries to return, 0 returns as many entries as possible
261     * @return a list of whatever you requested in the HQL
262     */
263    protected List<?> executeHqlQuery(String hql, Map<String, Object> params, int start, int limit) {
264       Query query = getSession().createQuery(hql);
265       query.setFirstResult(start);
266       if (limit > 0) {
267          query.setMaxResults(limit);
268       }
269       setParameters(query, params);
270       return query.list();
271    }

273    /***
274     * This is supported natively in Hibernate 3.2.x and up<br/>
275     * sets the parameters correctly for hibernate 3.1
276     * 
277     * @param query
278     * @param params
279     */
280    protected void setParameters(Query query, Map<String, Object> params) {
281       for (Entry<String, Object> entry : params.entrySet()) {
282          String name = entry.getKey();
283          Object param = entry.getValue();
284          if (param.getClass().isArray()) {
285             query.setParameterList(name, (Object[]) param);
286          } else {
287             query.setParameter(name, param);
288          }
289       }
290    }

304    /***
305     * Use hibernate metadata to get the id value
306     */
307    protected Serializable baseGetIdValue(Object object) {
308       Class<?> type = findClass(object);
309       Serializable idValue = null;
310       ClassMetadata classmeta = getSessionFactory().getClassMetadata(type);
311       if (classmeta != null) {
312          if (classmeta.hasIdentifierProperty()) {
313             idValue = classmeta.getIdentifier(object, EntityMode.POJO);
314          }
315       } else {
316          throw new IllegalArgumentException("Could not get classmetadata for this object, it may not be persistent: " + object);
317       }
318       return idValue;
319    }

405    /***
406     * This does a nice bit of exception handling for us and verifies that
407     * this class is valid to perform a DAO operation with
408     * @param type class type of the persistent object to check
409     * @return A valid entityClass type resolved to be the same as
410     * the ones usable by this DAO
411     */
412    protected Class<?> checkClass(Class<?> type) {
413       if (type == null) {
414          throw new NullPointerException("type cannot be null");
415       }
416 
417       if (classes == null) {
418          throw new NullPointerException("persistent classes must be set");
419       }
420 
421       for (Iterator<Class<?>> i = classes.iterator(); i.hasNext();) {
422          Class<?> concrete = (Class<?>) i.next();
423          if (concrete.isAssignableFrom(type)) {
424             return concrete;
425          }
426       }
427       throw new IllegalArgumentException("Could not resolve this class " + 
428             type + " as part of the set of persistent objects: " +
429             classes.toString());
430    }

481    private Map<Class<?>, ReadInterceptor> readInterceptors = new ConcurrentHashMap<Class<?>, ReadInterceptor>();
482    private Map<Class<?>, WriteInterceptor> writeInterceptors = new ConcurrentHashMap<Class<?>, WriteInterceptor>();
483    /***
484     * Adds the provided interceptor to the current set of interceptors
485     * @param interceptor
486     */
487    public void addInterceptor(DaoOperationInterceptor interceptor) {
488       if (interceptor != null) {
489          Class<?> type = interceptor.interceptType();
490          if (type != null) {
491             if (ReadInterceptor.class.isAssignableFrom(interceptor.getClass())) {
492                readInterceptors.put(type, (ReadInterceptor) interceptor);
493             }
494             if (WriteInterceptor.class.isAssignableFrom(interceptor.getClass())) {
495                writeInterceptors.put(type, (WriteInterceptor) interceptor);
496             }
497          }
498       }
499    }
500    /***
501     * Removes the provided interceptor from the current set of interceptors
502     * @param interceptor
503     */
504    public void removeInterceptor(DaoOperationInterceptor interceptor) {
505       if (interceptor != null) {
506          Class<?> type = interceptor.interceptType();
507          if (type != null) {
508             if (ReadInterceptor.class.isAssignableFrom(interceptor.getClass())) {
509                readInterceptors.remove(type);
510             }
511             if (WriteInterceptor.class.isAssignableFrom(interceptor.getClass())) {
512                writeInterceptors.remove(type);
513             }
514          }
515       }
516    }
517    /***
518     * Makes the given interceptor the only active interceptor,
519     * removes all others and adds only this one
520     * @param interceptor
521     */
522    public void setInterceptor(DaoOperationInterceptor interceptor) {
523       if (interceptor != null) {
524          readInterceptors.clear();
525          writeInterceptors.clear();
526          addInterceptor(interceptor);
527       }
528    }

583    /* (non-Javadoc)
584     * @see org.sakaiproject.genericdao.api.GenericDao#findById(java.lang.Class, java.io.Serializable)
585     */
586    @SuppressWarnings("unchecked")
587    public <T> T findById(Class<T> type, Serializable id) {
588       checkClass(type);
589       if (id == null) {
590          throw new IllegalArgumentException("id must be set to find persistent object");
591       }
592 
593       T entity = null;
594       // check cache first
595       String key = id.toString();
596       String cacheName = getCacheName(type);
597       if (getCacheProvider().exists(cacheName, key)) {
598          entity = (T) getCacheProvider().get(cacheName, key);
599       } else {
600          // not in cache so go to the DB
601    
602          // before interceptor
603          String operation = "findById";
604          beforeRead(operation, type, new Serializable[] {id}, null);
605    
606          entity = baseFindById(type, id);
607    
608          // now put the item in the cache
609          getCacheProvider().put(cacheName, key, entity);
610    
611          // after interceptor
612          afterRead(operation, type, new Serializable[] {id}, null, new Object[] {entity});
613       }
614       return entity;
615    }

618     * @see org.sakaiproject.genericdao.api.modifiers.BasicModifier#create(java.lang.Object)
619     */
620    public void create(Object object) {
621       Class<?> type = findClass(object);
622       checkClass(type);
623 
624       String operation = "create";
625       beforeWrite(operation, type, null, new Object[] {object});
626 
627       Serializable idValue = baseCreate(type, object);
628 
629       // clear the search caches
630       getCacheProvider().clear(getSearchCacheName(type));
631 
632       afterWrite(operation, type, new Serializable[] {idValue}, new Object[] {object}, 1);
633    }

635    /* (non-Javadoc)
636     * @see org.sakaiproject.genericdao.api.modifiers.BasicModifier#update(java.lang.Object)
637     */
638    public void update(Object object) {
639       Class<?> type = findClass(object);
640       checkClass(type);
641 
642       Serializable idValue = baseGetIdValue(object);
643       if (idValue == null) {
644          throw new IllegalArgumentException("Could not get an id value from the supplied object, cannot update without an id: " + object);
645       }
646 
647       String operation = "update";
648       beforeWrite(operation, type, new Serializable[] {idValue}, new Object[] {object});
649 
650       baseUpdate(type, idValue, object);
651 
652       // clear the search caches
653       getCacheProvider().clear(getSearchCacheName(type));
654       // clear the cache entry since this was updated
655       String key = idValue.toString();
656       String cacheName = getCacheName(type);
657       getCacheProvider().remove(cacheName, key);
658 
659       afterWrite(operation, type, new Serializable[] {idValue}, new Object[] {object}, 1);
660    }
