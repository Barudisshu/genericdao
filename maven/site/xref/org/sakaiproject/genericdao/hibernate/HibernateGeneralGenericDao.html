102    public <T> int countAll(Class<T> type) {
103       checkClass(type);
104       int count = 0;
105 
106       // check the cache first
107       boolean usedCache = false;
108       String searchCacheName = getSearchCacheName(type);
109       String cacheKey = "countAll::" + type.getName();
110       if (getCacheProvider().exists(searchCacheName, cacheKey)) {
111          Integer iCount = (Integer) getCacheProvider().get(searchCacheName, cacheKey);
112          if (iCount != null) {
113             count = iCount.intValue();
114             usedCache = true;
115          }
116       }
117 
118       if (! usedCache) {
119          count = baseCountAll(type);
120 
121          // cache the id results for the search
122          getCacheProvider().put(searchCacheName, cacheKey, Integer.valueOf(count));
123       }
124       return count;
125    }

127    public <T> void deleteSet(Class<T> type, Serializable[] ids) {
128       checkClass(type);
129       if (ids.length > 0) {
130          String operation = "deleteSet";
131          beforeWrite(operation, type, ids, null);
132    
133          int changes = baseDeleteSet(type, ids);
134    
135          afterWrite(operation, type, ids, null, changes);
136    
137          // clear all removed items from the cache
138          String cacheName = getCacheName(type);
139          for (int i = 0; i < ids.length; i++) {
140             if (ids[i] != null) {
141                String key = ids[i].toString();
142                getCacheProvider().remove(cacheName, key);
143             }
144          }
145          // clear the search caches
146          getCacheProvider().clear(getSearchCacheName(type));
147       }
148    }

150    public <T> void saveSet(Set<T> entities) {
151       if (entities == null || entities.isEmpty()) {
152          System.out.println("WARN: Empty list of entities for saveSet, nothing to do...");
153       } else {
154          Class<?> type = checkEntitySet(entities);
155    
156          String operation = "saveSet";
157          beforeWrite(operation, type, null, entities.toArray());
158    
159          int changes = baseSaveSet(type, entities);
160    
161          afterWrite(operation, type, null, entities.toArray(), changes);
162    
163          // clear all saved items from the cache
164          String cacheName = getCacheName(type);
165          for (T t : entities) {
166             Object id = baseGetIdValue(t);
167             if (id != null) {
168                String key = id.toString();
169                getCacheProvider().remove(cacheName, key);
170             }
171          }
172          // clear the search caches
173          getCacheProvider().clear(getSearchCacheName(type));
174       }
175    }

177    public <T> void deleteSet(Set<T> entities) {
178       if (entities.size() > 0) {
179          Class<?> type = checkEntitySet(entities);
180          List<Object> ids = new ArrayList<Object>();
181          for (T t : entities) {
182             Object id = baseGetIdValue(t);
183             if (id != null) {
184                ids.add(id);
185             }
186          }
187          deleteSet(type, ids.toArray(new Serializable[ids.size()]));
188       }
189    }

211    /***
212     * Validates the class type and the list of entities before performing
213     * a batch operation (throws IllegalArgumentException)
214     * 
215     * @param entities a Set of persistent entities, should all be of the same type
216     */
217    protected Class<?> checkEntitySet(Set<?> entities) {
218       Class<?> entityClass = null;
219       Iterator<?> it = entities.iterator();
220       while(it.hasNext()) {
221          Object entity = it.next();
222          if (entityClass == null) {
223             entityClass = (Class<?>) findClass(entity);
224          }
225          if (! checkClass(entityClass).isInstance(entity)) {
226             throw new IllegalArgumentException("Entity set item " +
227                   entity.toString() + " is not of type: " + entityClass +
228                   ", the type is: " + entity.getClass() +
229             " (All items must be of consistent persistent type)");
230          }
231       }
232       return entityClass;
233    }
