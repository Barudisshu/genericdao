43     /***
44      * Build the Criteria object here to reduce code duplication
45      * @param entityClass
46      * @param search a Search object (possibly only partially complete)
47      * @return a DetachedCriteria object
48      */
49     private DetachedCriteria buildCriteria(Class<?> entityClass, Search search) {
50        // Checks to see if the required params are set and throws exception if not
51        if (search == null) {
52           throw new IllegalArgumentException("search cannot be null");
53        }
54  
55        // Build the criteria object
56        DetachedCriteria criteria = DetachedCriteria.forClass(entityClass);
57  
58        // Only add in restrictions if there are some to add
59        if (search.getRestrictions() != null && search.getRestrictions().length > 0) {
60           Junction junction = Expression.conjunction(); // AND
61           if (! search.conjunction) {
62              // set to use disjunction
63              junction = Expression.disjunction(); // OR
64           }
65           criteria.add(junction);
66     
67           // put in the restrictions
68           for (int i = 0; i < search.getRestrictions().length; i++) {
69              String property = search.getRestrictions()[i].property;
70              Object value = search.getRestrictions()[i].value;
71              if (property == null || value == null) {
72                 throw new IllegalArgumentException("restrictions property and value cannot be null or empty");            
73              }
74              if (value.getClass().isArray()) {
75                 // special handling for "in" type comparisons
76                 Object[] objectArray = (Object[]) value;
77                 if (objectArray.length == 1) {
78                    value = objectArray[0];
79                 } else if (objectArray.length > 1) {
80                    if (Restriction.NOT_EQUALS == search.getRestrictions()[i].comparison) {
81                       junction.add( Restrictions.not( Restrictions.in(property, objectArray) ) );
82                    } else {
83                       junction.add(Restrictions.in(property, objectArray));
84                    }
85                 } else {
86                    // do nothing for now, this is slightly invalid but not worth dying over
87                 }
88              }
89     
90              if (! value.getClass().isArray()) {
91                 switch (search.getRestrictions()[i].comparison) {
92                 case Restriction.EQUALS:
93                    junction.add(Restrictions.eq(property, value));
94                    break;
95                 case Restriction.GREATER:
96                    junction.add(Restrictions.gt(property, value));
97                    break;
98                 case Restriction.LESS:
99                    junction.add(Restrictions.lt(property, value));
100                   break;
101                case Restriction.LIKE:
102                   junction.add(Restrictions.like(property, value));
103                   break;
104                case Restriction.NULL:
105                   junction.add(Restrictions.isNull( property ));
106                   break;
107                case Restriction.NOT_NULL:
108                   junction.add(Restrictions.isNotNull( property ));
109                   break;
110                case Restriction.NOT_EQUALS:
111                   junction.add(Restrictions.ne(property, value));
112                   break;
113                }
114             }
115          }
116       }
117 
118       // handle the sorting (sort param can be null for no sort)
119       if (search.getOrders() != null) {
120          for (int i = 0; i < search.getOrders().length; i++) {
121             if (search.getOrders()[i].ascending) {
122                criteria.addOrder(
123                      org.hibernate.criterion.Order.asc( search.getOrders()[i].property ));
124             } else {
125                criteria.addOrder(
126                      org.hibernate.criterion.Order.desc( search.getOrders()[i].property ));
127             }
128          }
129       }
130 
131       return criteria;
132    }

140    @SuppressWarnings("unchecked")
141    protected <T> long baseCountBySearch(Class<T> type, Search search) {
142       DetachedCriteria criteria = buildCriteria(type, search);
143       criteria.setProjection(Projections.rowCount());
144       List<Number> l = getHibernateTemplate().findByCriteria(criteria);
145       return l.get(0).longValue();
146    }

152    protected <T> List<T> baseFindBySearch(Class<T> type, Search search) {
153       DetachedCriteria criteria = buildCriteria(type, search);
154       List<T> items = getHibernateTemplate().findByCriteria(criteria, 
155             Long.valueOf(search.getStart()).intValue(), 
156             Long.valueOf(search.getLimit()).intValue());
157       // TODO need to figure out how to force persistent objects to be transitive
158       return items;
159    }

165       T item = null;

177    public <T> long countBySearch(Class<T> type, Search search) {
178       checkClass(type);
179       if (search == null) {
180           throw new IllegalArgumentException("search cannot be null");
181       }
182       long count = 0;
183 
184       // check the cache first
185       boolean usedCache = false;
186       String searchCacheName = getSearchCacheName(type);
187       String cacheKey = "countBySearch::" + type.getName() + ":" + search.toString();
188       if (getCacheProvider().exists(searchCacheName, cacheKey)) {
189          Long lCount = (Long) getCacheProvider().get(searchCacheName, cacheKey);
190          if (lCount != null) {
191             count = lCount.longValue();
192             usedCache = true;
193          }
194       }
195 
196       if (! usedCache) {
197          count = baseCountBySearch(type, search);
198 
199          // cache the id results for the search
200          getCacheProvider().put(searchCacheName, cacheKey, Long.valueOf(count));
201       }
202       return count;
203    }

205    @SuppressWarnings("unchecked")
206    public <T> List<T> findBySearch(Class<T> type, Search search) {
207       checkClass(type);
208       if (search == null) {
209           throw new IllegalArgumentException("search cannot be null");
210       }
211       List<T> results = new ArrayList<T>();
212 
213       // check the cache first
214       boolean usedCache = false;
215       String cacheName = getCacheName(type);
216       String searchCacheName = getSearchCacheName(type);
217       String cacheKey = "findBySearch::" + type.getName() + ":" + search.toString();
218       if (getCacheProvider().exists(searchCacheName, cacheKey)) {
219          String[] resultIds = (String[]) getCacheProvider().get(searchCacheName, cacheKey);
220          if (resultIds != null) {
221             for (int i = 0; i < resultIds.length; i++) {
222                if (! getCacheProvider().exists(cacheName, resultIds[i])) {
223                   usedCache = false;
224                   break;
225                }
226                T entity = (T) getCacheProvider().get(cacheName, resultIds[i]);
227                results.add(entity);
228             }
229             usedCache = true;
230          }
231       }
232 
233       if (! usedCache) {
234          String operation = "findBySearch";
235          beforeRead(operation, type, null, search);
236 
237          results = baseFindBySearch(type, search);
238 
239          // run through the returned items for the interceptor and for caching
240          List<String> keys = new ArrayList<String>();
241          for (T entity : results) {
242             Object id = baseGetIdValue(entity);
243             // cache each returned item
244             String key = id.toString();
245             keys.add(key);
246             getCacheProvider().put(cacheName, key, entity);
247          }
248          // cache the id results for the search
249          String[] ids = keys.toArray(new String[keys.size()]);
250          getCacheProvider().put(searchCacheName, cacheKey, ids);
251          // call the after interceptor
252          afterRead(operation, type, ids, search, results.toArray(new Object[results.size()]));
253       }
254       return results;
255    }

257    @SuppressWarnings("unchecked")
258    public <T> T findOneBySearch(Class<T> type, Search search) {
259       checkClass(type);
260       if (search == null) {
261           throw new IllegalArgumentException("search cannot be null");
262       }
263       T entity = null;
264 
265       // check the cache first
266       boolean usedCache = false;
267       String cacheName = getCacheName(type);
268       String searchCacheName = getSearchCacheName(type);
269       String cacheKey = "findOneBySearch::" + type.getName() + ":" + search.toString();
270       if (getCacheProvider().exists(searchCacheName, cacheKey)) {
271          usedCache = true;
272          String id = (String) getCacheProvider().get(searchCacheName, cacheKey);
273          if (id != null) {
274             if (getCacheProvider().exists(cacheName, id)) {
275                entity = (T) getCacheProvider().get(cacheName, id);
276             }
277          }
278       }
279 
280       if (! usedCache) {
281          String operation = "findOneBySearch";
282          beforeRead(operation, type, null, search);
283 
284          search.setLimit(1); // only return 1 item
285 
286          entity = baseFindOneBySearch(type, search);
287 
288          String key = null;
289          if (entity != null) {
290             Serializable id = baseGetIdValue(entity);
291             afterRead(operation, type, new Serializable[] {id}, search, new Object[] {entity});
292 
293             if (id != null) {
294                // cache the entity
295                key = id.toString();
296                getCacheProvider().put(cacheName, key, entity);
297             }
298          }
299          // cache the search result
300          getCacheProvider().put(searchCacheName, cacheKey, key);
301       }
302       return entity;
303    }

376    /***
377     * @deprecated
378     */
379    @SuppressWarnings("unchecked")
380    public List findByProperties(Class entityClass, String[] objectProperties, Object[] values,
381          int[] comparisons, String[] sortProperties, int firstResult, int maxResults) {
382       if (objectProperties.length != values.length || values.length != comparisons.length) {
383          throw new IllegalArgumentException("All input arrays must be the same size");
384       }
385       Search search = new Search();
386       for (int i = 0; i < values.length; i++) {
387          search.addRestriction( new Restriction(objectProperties[i], values[i], comparisons[i]) );
388       }
389       if (sortProperties != null) {
390          for (int i = 0; i < sortProperties.length; i++) {
391             int location = sortProperties[i].indexOf(" ");
392             String property = sortProperties[i];
393             if (location > 0) {
394                property = sortProperties[i].substring(0, location);
395             }
396             Order order = null;
397             if (sortProperties[i].endsWith(ByPropsFinder.DESC)) {
398                order = new Order(property, false);
399             } else {
400                order = new Order(property, true);
401             }
402             search.addOrder( order );
403          }
404       }
405       search.setStart(firstResult);
406       search.setLimit(maxResults);
407       return findBySearch(entityClass, search);
408    }
