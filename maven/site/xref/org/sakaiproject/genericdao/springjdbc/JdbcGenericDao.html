104         } else if (DatabaseTranslator.DBTYPE_DERBY.equalsIgnoreCase(databaseType)) {
105             this.databaseType = DatabaseTranslator.DBTYPE_DERBY;
106         } else if (DatabaseTranslator.DBTYPE_HSQLDB.equalsIgnoreCase(databaseType)) {
107             this.databaseType = DatabaseTranslator.DBTYPE_HSQLDB;
108         } else if (DatabaseTranslator.DBTYPE_MSSQL.equalsIgnoreCase(databaseType)) {
109             this.databaseType = DatabaseTranslator.DBTYPE_MSSQL;
110         } else if (DatabaseTranslator.DBTYPE_MYSQL.equalsIgnoreCase(databaseType)) {

114         } else if (DatabaseTranslator.DBTYPE_POSTGRES.equalsIgnoreCase(databaseType)) {
115             this.databaseType = DatabaseTranslator.DBTYPE_POSTGRES;
116         } else {
117             this.databaseType = databaseType.toUpperCase();
118         }

233      * @param dataSource any non-spring {@link DataSource}

238     public void setNonSpringDataSource(DataSource dataSource, boolean threadBound) {
239         if (dataSource == null) {
240             throw new IllegalArgumentException("dataSource cannot be null");
241         }
242         if (dataSource instanceof SmartDataSource) {
243             setDataSource(dataSource);
244         } else {
245             // wrap it up
246             DataSource wrapper = dataSource;
247             if (threadBound) {
248                 wrapper = new ThreadboundConnectionsDataSourceWrapper(wrapper);
249             }
250             wrapper = new SmartDataSourceWrapper(wrapper);
251             setDataSource( wrapper );
252         }

261     /***
262      * This tells the DAO about your persistent classes and their associated tables,
263      * it is the major configuration path for the DAO
264      * @param dataMappers a list of all {@link DataMapper}s that this DAO uses,
265      * ideally this is no more than a few per DAO but it should include all tables which are directly linked
266      */
267     public void setDataMappers(List<DataMapper> dataMappers) {
268         if (this.dataMappers == null) {
269             this.dataMappers = new ConcurrentHashMap<Class<?>, DataMapper>();
270             this.classes = new Vector<Class<?>>();
271         }
272         for (DataMapper dataMapper : dataMappers) {
273             Class<?> type = dataMapper.getPersistentType();
274             this.dataMappers.put(type, dataMapper);
275             this.classes.add(type);
276             getNamesRecord(type); // prime the names record
277         }
278     }

280     /***
281      * Get a data mapper for a specific class,
282      * will always return a data mapper
283      */
284     protected DataMapper getDataMapper(Class<?> type) {
285         DataMapper dm = dataMappers.get(type);
286         if (dm == null && classes.contains(type)) {
287             // make a Simple DM, this assumes the necessary tables already exist
288             dm = new SimpleDataMapper(type);
289             dataMappers.put(type, dm); // place it in the map
290         }
291         if (dm == null) {
292             throw new IllegalArgumentException("type is not a persistent class type: " + type);
293         }
294         return dm;
295     }

297     // names mapping cache
298     private Map<Class<?>, NamesRecord> namesRecordsCache = new ConcurrentHashMap<Class<?>, NamesRecord>();
299     /***
300      * Get a names mapping for a specific class type,
301      * uses caching, will always return a mapping
302      */
303     public NamesRecord getNamesRecord(Class<?> type) {
304         NamesRecord nr = namesRecordsCache.get(type);
305         if (nr == null) {
306             // get a names record from the data mapper
307             DataMapper dm = getDataMapper(type);
308             nr = dm.getPropertyToColumnNamesMapping();
309             if (nr != null) {
310                 nr.setIdentifierProperty( findIdProperty(type) );
311                 if (nr.getForeignKeyPropertyNames().isEmpty()) {
312                     // check for foreign keys and add them (only complete fields though)
313                     Map<String, Class<?>> types = ReflectUtils.getInstance().getFieldTypes(type, FieldsFilter.COMPLETE);
314                     for (Entry<String, Class<?>> entry : types.entrySet()) {
315                         // special handling for foreign keys identified by persistent types inside this object
316                         String property = entry.getKey();
317                         Class<?> pType = entry.getValue();
318                         if (getPersistentClasses().contains(pType)) {
319                             // this is another persistent object so this must be a foreign key
320                             String pId = findIdProperty(pType);
321                             String fkProp = property + "." + pId;
322                             String column = nr.getColumnForProperty(property);
323                             if (column != null) {
324                                 nr.setForeignKeyMapping(fkProp, column);
325                             }
326                         }
327                     }
328                 }
329                 namesRecordsCache.put(type, nr);
330             }
331         }
332         if (nr == null) {
333             boolean usePropertyNamesForColumns = false;
334             DataMapper dm = getDataMapper(type);
335 
336             ReflectUtils reflectUtils = ReflectUtils.getInstance();
337             // try to get the mapping from the class using annotations
338             ClassFields<?> classFields = reflectUtils.analyzeClass(type);
339             if (classFields.getClassAnnotations().contains(PersistentColumnMappingPolicy.class)) {
340                 for (Annotation classAnnote : classFields.getClassAnnotations()) {
341                     if (PersistentColumnMappingPolicy.class.equals(classAnnote.annotationType())) {
342                         MappingPolicy mp = ((PersistentColumnMappingPolicy)classAnnote).policy();
343                         if (MappingPolicy.FIELD_NAMES.equals(mp)) {
344                             usePropertyNamesForColumns = true;
345                         } else if (MappingPolicy.UPPER_UNDERSCORES.equals(mp)) {
346                             usePropertyNamesForColumns = false;
347                         }
348                     }
349                     if (dm instanceof SimpleDataMapper) {
350                         // override the setting
351                         ((SimpleDataMapper)dm).setUsePropertyNamesForColumns(usePropertyNamesForColumns);
352                     }
353                 }
354             } else {
355                 // no annotation so get the setting from the data mapper
356                 if (dm instanceof SimpleDataMapper) {
357                     usePropertyNamesForColumns = ((SimpleDataMapper)dm).isUsePropertyNamesForColumns();
358                 }
359             }
360 
361             // create a names mapping using reflection
362             nr = new NamesRecord();
363             Map<String, Class<?>> types = reflectUtils.getFieldTypes(type, FieldsFilter.COMPLETE);
364             for (String property : types.keySet()) {
365                 String column = property;
366                 // check for transient annotation
367                 try {
368                     Annotation annotation = classFields.getFieldAnnotation(PersistentTransient.class, property);
369                     if (annotation != null) {
370                         // skip this one
371                         continue;
372                     }
373                 } catch (FieldnameNotFoundException e) {
374                     // nothing to do
375                 }
376                 if (! usePropertyNamesForColumns) {
377                     column = BasicTranslator.makeDBNameFromCamelCase(property);
378                 }
379                 // check for annotation override to column name
380                 try {
381                     PersistentColumnName annotation = (PersistentColumnName) classFields.getFieldAnnotation(PersistentColumnName.class, property);
382                     if (annotation != null) {
383                         column = annotation.value();
384                     }
385                 } catch (FieldnameNotFoundException e) {
386                     // nothing to do
387                 }
388                 nr.setNameMapping(property, column);
389                 // special handling for foreign keys identified by persistent types inside this object
390                 Class<?> pType = types.get(property);
391                 if (getPersistentClasses().contains(pType)) {
392                     // this is another persistent object so this must be a foreign key
393                     String pId = findIdProperty(pType);
394                     String fkProp = property + "." + pId;
395                     nr.setForeignKeyMapping(fkProp, column);
396                 }
397             }
398             // add in the special id marker and make sure the id is set right
399             nr.setIdentifierProperty( findIdProperty(type) );
400             namesRecordsCache.put(type, nr);
401             if (dm instanceof SimpleDataMapper) {
402                 // put this NamesRecord back into the DataMapper
403                 ((SimpleDataMapper)dm).setNamesRecord(nr);
404             }
405         }
406         return nr;
407     }

451     /***
452      * Initialize the persistent classes
453      */
454     protected void initPersistentClasses() {
455         // init the list of classes and mappers related to them and execute DDL if needed
456         for (Class<?> type : getPersistentClasses()) {
457             DataMapper dm = getDataMapper(type);
458             if (autoDDL) {
459                 InputStream ddlIS = null;
460                 // try to get DDL from the mapper first
461                 String ddl = dm.generateDDL(getDatabaseType());
462                 if ( ddl == null || "".equals(ddl) ) {
463                     if (dm instanceof SimpleDataMapper) {
464                         // try loading from a file if set
465                         SimpleDataMapper sdm = (SimpleDataMapper) dm;
466                         String filepath = sdm.getDbTypeToFilename().get(getDatabaseType());
467                         if (filepath != null) {
468                             // cleanup filename
469                             if (filepath.startsWith("/")) {
470                                 filepath = filepath.substring(1);
471                             }
472                             // try looking in the classloader of the thread first
473                             ddlIS = getInputStreamFromClassloader(Thread.currentThread().getContextClassLoader(), filepath);
474                             if (ddlIS == null) {
475                                 // next try the classloader for this DAO
476                                 ddlIS = getInputStreamFromClassloader(this.getClass().getClassLoader(), filepath);
477                             }
478                             if (ddlIS == null) {
479                                 // next try the classloader for persistent type
480                                 ddlIS = getInputStreamFromClassloader(type.getClassLoader(), filepath);
481                             }
482                             if (ddlIS == null) {
483                                 // we got a filename but did not find the file contents, we need to die
484                                 throw new IllegalArgumentException("Could not find find DDL file resource ("+filepath+") for DB ("+getDatabaseType()+") in any searched classloader, cannot execute DDL");
485                             }
486                         }
487                     } else {
488                         // nothing to do here: we have a blank ddl and this is not a simple mapper so just continue on -AZ
489                     }
490                 } else {
491                     // turn DDL into an IS
492                     ddlIS = new ByteArrayInputStream(ddl.getBytes());
493                 }
494                 if (ddlIS != null) {
495                     // execute the ddl
496                     executeDDLforType(ddlIS, type);
497                 }
498             }
499         }
500     }

928      * Execute a DDL (database definition language) script, this will execute
929      * a set of DDL commands which are fed in via an InputStream<br/>
930      * The first non-comment ('--') line will be run, and if successful, 
931      * all other non-comment lines will be run. SQL statements may be on 
932      * multiple lines but must have ';' terminators.<br/>
933      * <br/>
934      * <b>NOTE:</b> The script should be located in the current ClassLoader,
935      * in other words, it must be a visible resource that is packaged with
936      * the code that is running this method<br/>
937      * Recommended (Example) usage:<br/>
938      * 1) Place the script into the src folder of your service impl project like so:
939      * <code>impl/src/sql/oracle/myscript.sql</code><br/>
940      * 2) Load the sql files into the jar by setting up the maven pom like so:
941      * <xmp>
942 <resources>
943 ...
944    <resource>
945       <directory>${basedir}/src/sql</directory>
946       <includes>
947          <include>** /*.sql</include>
948       </includes>
949    </resource>
950 ...
951 </resources></xmp> 
952      * <b>Note:</b> remove the extra space between "**" and "/*.sql"<br/>
953      * 3) Execute the ddl script in the init method of your DAO (when it first starts up):
954      * <code>ClassLoader loader = this.getClass().getClassLoader();</code><br/>
955      * <code>String ddlResource = getDatabaseType() + "/myscript.sql";</code><br/>
956      * <code>InputStream stream = loader.getResourceAsStream(ddlResource);</code><br/>
957      * <code>executeDDL(stream);</code><br/>
958      * 
959      * @param sqlDDL the sql ddl commands to execute
960      * @throws IllegalArgumentException if ddl is invalid and cannot be executed
961      */
962     public void executeDDL(InputStream sqlDDL) {
963         if (sqlDDL == null) {
964             throw new IllegalArgumentException("sqlDDL cannot be null");
965         }
966 
967         executeDDLforType(sqlDDL, null);
968     }

970     private void executeDDLforType(InputStream sqlDDL, Class<?> type) {
971         // Now run the DDL commands if possible
972         try {
973             if (isAutoCommitDDL()) {
974                 commitTransaction(); // start the transaction
975             }
976             BufferedReader r = new BufferedReader(new InputStreamReader(sqlDDL));
977             try {
978                 // read the first line, skipping any '--' comment lines
979                 boolean firstLine = true;
980                 StringBuffer buf = new StringBuffer();
981                 for (String line = r.readLine(); line != null; line = r.readLine()) {
982                     line = line.trim();
983                     if (line.startsWith("--")) continue;
984                     if (line.length() == 0) continue;
985 
986                     // add the line to the buffer
987                     buf.append(' ');
988                     buf.append(line);
989 
990                     // process if the line ends with a ';'
991                     boolean process = line.endsWith(";");
992 
993                     if (!process) continue;
994 
995                     // remove trailing ';'
996                     buf.setLength(buf.length() - 1);
997 
998                     String ddl = buf.toString().trim();
999                     // FIXME do replacements even if we do not know the type
1000                     if (type != null) {
1001                         // handle ddl replacements
1002                         ddl = handleTypeReplacements(type, ddl);
1003                     }
1004 
1005                     // run the first line as the test - if it fails, we are done
1006                     if (firstLine) {
1007                         firstLine = false;
1008                         try {
1009                             if (showSQL) {
1010                                 logInfo("DDL="+ddl);
1011                             }
1012                             getSpringJdbcTemplate().execute(ddl);
1013                         } catch (DataAccessException e) {
1014                             //log.info("Could not to execute first DDL ("+ddl+"), skipping the rest");
1015                             logInfo("Could not execute first DDL line, skipping the rest: " + e.getMessage() + ":" + e.getCause());
1016                             //e.printStackTrace();
1017                             return;
1018                         }
1019                     } else {
1020                         // run other lines, until done - any one can fail (we will report it)
1021                         try {
1022                             if (showSQL) {
1023                                 logInfo("DDL="+ddl);
1024                             }
1025                             getSpringJdbcTemplate().execute(ddl);
1026                         } catch (DataAccessException e) {
1027                             throw new IllegalArgumentException("Failed while executing ddl: " + e.getMessage(), e);
1028                         }
1029                     }
1030                     if (isAutoCommitDDL()) {
1031                         commitTransaction();
1032                     }
1033 
1034                     // clear the buffer for next
1035                     buf.setLength(0);
1036                 }
1037             } catch (IOException any) {
1038                 throw new RuntimeException("Failure while processing DDL", any);
1039             } finally {
1040                 try {
1041                     r.close();
1042                 } catch (IOException any) {
1043                     //log.warn("Failure while closing DDL inputstream reader", any);
1044                 }
1045                 // close the connection used for this DDL
1046                 if (isAutoCommitDDL()) {
1047                     closeConnection();
1048                 }
1049             }
1050         } finally {
1051             try {
1052                 sqlDDL.close();
1053             } catch (IOException any) {
1054                 //log.warn("Failure while closing inputstream", any);
1055             }
1056         }
1057     }

1061     /***
1062      * Make SQL from a template
1063      * 
1064      * @param sqlTemplate pass in an SQL template (one of the BASIC_* ones or make your own),
1065      * the replacement names will be replaced with the replacement values
1066      * @param tableName the name of the table for this SQL
1067      * @param replacements a sequence of replacement names (probably {@link #COLUMNS}, {@link #WHERE}, etc.) and values,
1068      * alternating like so: name,value,name,value
1069      * @return the SQL with all replacements made
1070      */
1071     protected String makeSQL(String sqlTemplate, String tableName, String... replacements) {
1072         String sql = sqlTemplate;
1073         sql = sql.replace(StatementMapper.TABLE_NAME, tableName);
1074         for (int i = 0; i < replacements.length; i++) {
1075             if (replacements.length < i + 1) {
1076                 break;
1077             }
1078             sql = sql.replace(replacements[i], replacements[i+1]);
1079             i++;
1080         }
1081         // put in the select replacement last in case it was already replaced
1082         sql = sql.replace(StatementMapper.SELECT, "*");
1083         return sql;
1084     }

1086     /***
1087      * Handle the replacements of DDL constants in the supplied DDL with real values
1088      * based on the supplied type<br/>
1089      * The following keys will be replaced automatically:<br/>
1090      * {TABLENAME} - the value returned by {@link #getTableName()}<br/>
1091      * {ID} - the column name of the unique identifier<br/>
1092      * {TABLENAME:org.domain.MyClass} - the value returned by {@link #getTableName()} for the persistent type MyClass<br/>
1093      * {ID:org.domain.MyClass} - the column name of the unique identifier for the persistent type MyClass<br/>
1094      * {COLUMNNAME:propertyName} - the column name which maps to the propertyName<br/>
1095      * {IDSEQNAME} - (Oracle) a sequence name will be generated and inserted
1096      * based on the table name for use in generating IDs,
1097      * if you want to specify your own sequence name then you will lose
1098      * the ability to have the ID inserted into newly created objects<br/>
1099      * 
1100      * @param type a persistent class type
1101      * @param ddl database definition statements (can also be SQL)
1102      * @return the DDL string with the {vars} replaced
1103      */
1104     protected String handleTypeReplacements(Class<?> type, String ddl) {
1105         //StringBuilder sb = new StringBuilder(ddl);
1106         if (type == null || ddl == null) {
1107             throw new IllegalArgumentException("Type and ddl cannot be null");
1108         }
1109         String tableName = getTableNameFromClass(type);
1110         ddl = ddl.replace(DataMapper.DDL_TABLENAME, tableName);
1111         ddl = ddl.replace(DataMapper.DDL_ID_COLUMN, getIdColumn(type));
1112         if (DataMapper.DBTYPE_ORACLE.equals(getDatabaseType())) {
1113             ddl = ddl.replace(DataMapper.DDL_ID_SEQNAME, OracleTranslator.getOracleSeqName(tableName) );
1114         }
1115         // now check for the other types and handle them if needed
1116         if (ddl.contains(DataMapper.DDL_TABLENAME_TYPE_PREFIX)) {
1117             // resolve the classname for a tablename
1118             List<Class<?>> pClasses = getPersistentClasses();
1119             for (Class<?> pType : pClasses) {
1120                 // attempt to replace all parent tablenames
1121                 String tName = getTableNameFromClass(pType);
1122                 ddl = ddl.replace(DataMapper.DDL_TABLENAME_TYPE_PREFIX + pType.getName() + "}", tName);
1123                 ddl = ddl.replace(DataMapper.DDL_TABLENAME_TYPE_PREFIX + pType.getSimpleName() + "}", tName);
1124             }
1125         }
1126         if (ddl.contains(DataMapper.DDL_ID_TYPE_PREFIX)) {
1127             // resolve the classname for a tablename
1128             List<Class<?>> pClasses = getPersistentClasses();
1129             for (Class<?> pType : pClasses) {
1130                 // attempt to replace all parent tablenames
1131                 String pId = getIdColumn(pType);
1132                 ddl = ddl.replace(DataMapper.DDL_ID_TYPE_PREFIX + pType.getName() + "}", pId);
1133                 ddl = ddl.replace(DataMapper.DDL_ID_TYPE_PREFIX + pType.getSimpleName() + "}", pId);
1134             }
1135         }
1136         if (ddl.contains(DataMapper.DDL_COLUMN_PREFIX)) {
1137             // resolve the classname for a tablename
1138             NamesRecord nr = getNamesRecord(type);
1139             for (String property : nr.getPropertyNames()) {
1140                 // attempt to replace all property names
1141                 String column = nr.getColumnForProperty(property);
1142                 ddl = ddl.replace(DataMapper.DDL_COLUMN_PREFIX + property + "}", column);
1143             }
1144             // TODO support for columns by type?
1145         }
1146         return ddl;
1147     }

1195                 conn.setAutoCommit(previousAutocommit);
1196             } catch (SQLException e1) {
1197                 // nothing to do here but continue
1198             }
1199         }
1200         // removed: try-finally-releaseConnection(conn);
1201         return success;
