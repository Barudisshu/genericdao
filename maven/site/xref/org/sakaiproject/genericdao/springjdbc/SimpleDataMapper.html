160     private NamesRecord namesRecord = null;
161     /***
162      * (OPTIONAL)
163      * Set this to a map of the persistent object properties to database column names,
164      * String -> String,<br/>
165      * example: "id" -> "ID", "title" -> "ITEM_TITLE"<br/>
166      * defaults to autogenerated column names which are uppercased and underscored (as in example above)
167      * @see SimpleDataMapper#setNamesUsed(String[])
168      * @see SimpleDataMapper#setNamesRecord(NamesRecord)
169      */
170     public void setNamesMapping(Map<String, String> namesMapping) {
171         if (namesMapping != null && ! namesMapping.isEmpty()) {
172             namesRecord = new NamesRecord();
173             for (Entry<String, String> entry : namesMapping.entrySet()) {
174                 String key = entry.getKey();
175                 String dbName = entry.getValue();
176                 if (dbName != null && ! "".equals(dbName)) {
177                     namesRecord.setNameMapping(key, dbName);
178                 }
179             }
180         }
181     }
182     /***
183      * (OPTIONAL)
184      * Set this to an array of the persistent object properties which will be used for this mapping,
185      * anything not listed will be ignored when forming queries,
186      * this is the least work but depends on the more conventions and allows the least control <br/>
187      * NOTE: Be sure to set {@link #setUsePropertyNamesForColumns(boolean)} before calling this method <br/>
188      * Example: ["id", "title"],<br/>
189      * "id" -> "ID", "title" -> "ITEM_TITLE"<br/>
190      * defaults to autogenerated column names which are uppercased and underscored (as in example above)
191      * depending on the setting for {@link #isUsePropertyNamesForColumns()}
192      * @see SimpleDataMapper#setNamesMapping(Map)
193      * @see SimpleDataMapper#setNamesRecord(NamesRecord)
194      */
195     public void setNamesUsed(String[] namesUsed) {
196         if (namesUsed == null) {
197             throw new IllegalArgumentException("NamesRecord cannot be null");
198         }
199         namesRecord = new NamesRecord();
200         for (int i = 0; i < namesUsed.length; i++) {
201             String propertyName = namesUsed[i];
202             String columnName = propertyName;
203             if (! isUsePropertyNamesForColumns()) {
204                 columnName = BasicTranslator.makeDBNameFromCamelCase(propertyName);
205             }
206             namesRecord.setNameMapping(propertyName, columnName);
207         }
208     }

260     /***
261      * (REQUIRED/OPTIONAL)
262      * Sets the map of databaseTypeConstant -> file containing the DDL script,
263      * the DDL will be looked up and stored in the class and executed on service init<br/>
264      * This must be set if the table does not already exist but if it does then this is optional<br/>
265      * The first non-comment ('--') line will be run, and if successful, 
266      * all other non-comment lines will be run. SQL statements may be on 
267      * multiple lines but must have ';' terminators.<br/>
268      * Can use the {@link #makeDDLTypeMap(String, String[])} method to make this more easily <br/>
269      * The following keys will be replaced automatically:<br/>
270      * {TABLENAME} - the value returned by {@link #getTableName()}<br/>
271      * {ID} - the column name of the unique identifier<br/>
272      * {IDSEQNAME} - (Oracle) a sequence name will be generated and inserted
273      * based on the tablename for use in generating IDs,
274      * if you want to specify your own sequence name then you will lose
275      * the ability to have the ID inserted into newly created objects<br/>
276      * <b>NOTE:</b> the file must be in your jar/war/package so it can
277      * be located in the classloader,
278      * use something like this in your maven pom.xml resources tag:
279      * <xmp><resource>
280         <directory>${basedir}/src/sql</directory>
281         <includes>
282           <include>** /*.sql</include>
283         </includes>
284       </resource></xmp>
285      * Remove the space between ** and /* from the above sample<br/>
286      * For example: src/sql/mysql.sql, src/sql/mysql/myproject.ddl<br/>
287      * @see #setDBTypeToDDL(Map)
288      * @see #makeDDLTypeMap(String, String[])
289      */
290     public void setDBTypeToFile(Map<String, String> dbTypeToFile) {
291         if (dbTypeToFilename == null) {
292             dbTypeToFilename = new HashMap<String, String>();
293         }
294         if (dbTypeToFile != null) {
295             for (Entry<String, String> entry : dbTypeToFile.entrySet()) {
296                 String dbtype = entry.getKey().toUpperCase();
297                 String value = entry.getValue();
298                 if (value != null) {
299                     this.dbTypeToFilename.put(dbtype, value);
300                 }
301             }
302         }
303     }

351     /***
352      * This is a convenience method which assists in creating the ddl type map,
353      * it basically allows you to easily create the map if you have followed the convention
354      * of placing all your ddl files in folders which are equivalent to the name
355      * of the database type they are created for and named them all identically with
356      * a name which is similar to the 
357      * @param fileName the name of the file (e.g. users.sql)
358      * @param types the types to include in the map, use the constants from {@link DatabaseTranslator} (e.g. "derby")
359      * @return the map which can be given to the {@link #setDBTypeToFile(Map)} method
360      */
361     public static Map<String, String> makeDDLTypeMap(String fileName, String[] types) {
362         if (fileName == null || "".equals(fileName)) {
363             throw new IllegalArgumentException("fileName must not be null");
364         }
365         if (types == null || types.length == 0) {
366             throw new IllegalArgumentException("Must include at least one type in the types array");
367         }
368         HashMap<String, String> dbTypeToFile = new HashMap<String, String>();
369         for (String type : types) {
370             dbTypeToFile.put(type, type.toLowerCase() + File.separator + fileName);
371         }
372         return dbTypeToFile;
373     }

375     /***
376      * Convenience method which will take a filename and an array of DBTYPE String constants
377      * (from {@link DatabaseTranslator}, example {@link DatabaseTranslator#DBTYPE_MYSQL})
378      * and produce a map of standard DDL types to file paths <br/>
379      * Will create a map like so: dbType -> [prefixPath/]dbType/fileName
380      * 
381      * @param fileName the name of the sql file (e.g. myTableDDL.sql)
382      * @param dbTypes an array with 1 or more dbTypes which you have DDL files for
383      * @param prefixPath (optional) the prefix to append before the dbType and fileName (should not have a leading "/"),
384      * can be null or "" if npt used
385      * @return the map of dbType to file paths
386      * @throws IllegalArgumentException if the filename or dbTYpes are null or empty
387      */
388     public static Map<String, String> makeDDLMap(String fileName, String[] dbTypes, String prefixPath) {
389         if (fileName == null || "".equals(fileName)) {
390             throw new IllegalArgumentException("filename must be set and cannot be null");
391         }
392         if (dbTypes == null || dbTypes.length == 0) {
393             throw new IllegalArgumentException("at least one dbType must be set and included, cannot be null or empty");
394         }
395         HashMap<String, String> dbTypeToFile = new HashMap<String, String>();
396         if (prefixPath == null) {
397             prefixPath = "";
398         } else {
399             if (! prefixPath.endsWith(File.separator)) {
400                 prefixPath += File.separator;
401             }
402         }
403         for (int i = 0; i < dbTypes.length; i++) {
404             String type = dbTypes[i];
405             dbTypeToFile.put(type, prefixPath + type.toLowerCase() + File.separator + fileName);
406         }
407         return dbTypeToFile;
408     }
