99          public void setValues(PreparedStatement ps, int index) throws SQLException {
100             Object entity = entities.get(index);
101             Map<String, Object> data = makeMapFromEntity(entity);
102             for (int i = 0; i < dataKeys.size(); i++) {
103                 String column = dataKeys.get(i);

105                 // convert value if needed
106                 value = JDBCUtils.convertColumn(namesRecord, column, value);
107                 ps.setObject((i+1), value);
108             }
109         }

126     @SuppressWarnings("unchecked")
127     public <T> List<T> findAll(Class<T> type, int firstResult, int maxResults) {
128         checkClass(type);
129         String tableName = getTableNameFromClass(type);
130         String sql = makeSQL(getSelectTemplate(type), tableName, 
131                 StatementMapper.SELECT, tableName + ".*", StatementMapper.WHERE, "");
132         // handle limit/offset
133         if (firstResult > 0 || maxResults > 0) {
134             firstResult = firstResult < 0 ? 0 : firstResult;
135             maxResults = maxResults < 0 ? 0 : maxResults;
136             sql = getDatabaseTranslator().makeLimitQuery(sql, firstResult, maxResults, tableName);
137             getSpringJdbcTemplate().setMaxRows(firstResult + maxResults); // this limit is always ok to put in
138         }
139         if (showSQL) {
140             logInfo("SQL="+sql);
141         }
142         List<Map<String, Object>> rMap = getSpringJdbcTemplate().queryForList(sql);
143         getSpringJdbcTemplate().setMaxRows(0); // reset this to no limit
144         List<T> results = new ArrayList<T>();
145 
146         // put the results into objects
147         int counter = 0;
148         // SPECIAL handling for DERBY
149         boolean derby = BasicTranslator.DBTYPE_DERBY.equals(getDatabaseType());
150         for (Map<String, Object> data : rMap) {
151             if (derby) {
152                 // derby has to filter results after the fact... lame yes indeed
153                 if (counter < firstResult) {
154                     counter++;
155                     continue;
156                 }
157                 if (maxResults > 0 && results.size() >= maxResults) {
158                     break;
159                 }
160             }
161             T entity = makeEntityFromMap(type, data);
162             results.add(entity);
163             counter++;
164         }
165         return results;
166     }

186     protected <T> int baseSaveSet(Class<?> type, Set<T> entities) {
187         String idProp = getIdProperty(type);
188         List<String> keys = new ArrayList<String>();
189         NamesRecord nr = getNamesRecord(type);
190 
191         List<Object> newObjects = new ArrayList<Object>();
192         List<Object> existingObjects = new ArrayList<Object>();
193         for (Object object : entities) {
194             Object id = ReflectUtils.getInstance().getFieldValue(object, idProp);
195             if (id == null) {
196                 newObjects.add(object);
197             } else {
198                 existingObjects.add(object);            
199             }
200         }
201 
202         Map<String, Class<?>> types = ReflectUtils.getInstance().getFieldTypes(type);
203         StringBuilder update = new StringBuilder();
204         StringBuilder columns = new StringBuilder();
205         StringBuilder values = new StringBuilder();
206         int counter = 0;
207         for (String key : types.keySet()) {
208             if (idProp.equals(key)) {
209                 continue;
210             }
211             if (counter > 0) {
212                 update.append(',');
213                 // insert
214                 columns.append(',');
215                 values.append(',');
216             }
217             String column = nr.getColumnForProperty(key);
218             update.append(column);
219             update.append("=?");
220             keys.add(column);
221             // insert
222             columns.append(column);
223             values.append('?');
224             counter++;
225         }
226         // make and do inserts
227         int changes = 0;
228         if (newObjects.size() > 0) {
229             String sql = makeSQL(getInsertTemplate(type), getTableNameFromClass(type), 
230                     StatementMapper.COLUMNS, columns.toString(), StatementMapper.VALUES, values.toString());
231             if (showSQL) {
232                 logInfo("SQL="+sql+":\n BatchCreate="+keys);
233             }
234             getSpringJdbcTemplate().batchUpdate(sql, new MyPSS(keys, newObjects, nr));
235         }
236         // make and do updates
237         if (existingObjects.size() > 0) {
238             keys.add( nr.getColumnForProperty(idProp) );
239             String sql = makeSQL(getUpdateTemplate(type), getTableNameFromClass(type), 
240                     StatementMapper.UPDATE, update.toString(), StatementMapper.WHERE, "where " + getIdColumn(type) + " = ?");
241             // do the batch update
242             if (showSQL) {
243                 logInfo("SQL="+sql+":\n BatchUpdate="+keys);
244             }
245             getSpringJdbcTemplate().batchUpdate(sql, new MyPSS(keys, existingObjects, nr));
246         }
247         return changes;
248     }

253     protected <T> int baseDeleteSet(Class<T> type, Serializable[] ids) {
254         StringBuilder sb = new StringBuilder();
255         for (int i = 0; i < ids.length; i++) {
256             Object id = ids[i];
257             if (id != null) {
258                 if (i > 0) { sb.append(','); }
259                 sb.append('?');
260             }
261         }
262         String sql = makeSQL(getDeleteTemplate(type), getTableNameFromClass(type), 
263                 StatementMapper.WHERE, "where " + getIdColumn(type) + " in (" + sb + ")");
264         if (showSQL) {
265             logInfo("SQL="+sql+":\n BatchDelete="+ArrayUtils.arrayToString(ids));
266         }
267         int changes = getSpringJdbcTemplate().update(sql, ids);
268         return changes;
269     }

277     public <T> int countAll(Class<T> type) {
278         checkClass(type);
279         int count = 0;
280 
281         // check the cache first
282         boolean usedCache = false;
283         String searchCacheName = getSearchCacheName(type);
284         String cacheKey = "countAll::" + type.getName();
285         if (getCacheProvider().exists(searchCacheName, cacheKey)) {
286             Integer iCount = (Integer) getCacheProvider().get(searchCacheName, cacheKey);
287             if (iCount != null) {
288                 count = iCount.intValue();
289                 usedCache = true;
290             }
291         }
292 
293         if (! usedCache) {
294             count = baseCountAll(type);
295 
296             // cache the id results for the search
297             getCacheProvider().put(searchCacheName, cacheKey, Integer.valueOf(count));
298         }
299         return count;
300     }

302     public <T> void deleteSet(Class<T> type, Serializable[] ids) {
303         checkClass(type);
304         if (ids.length > 0) {
305             String operation = "deleteSet";
306             beforeWrite(operation, type, ids, null);
307 
308             int changes = baseDeleteSet(type, ids);
309 
310             afterWrite(operation, type, ids, null, changes);
311 
312             // clear all removed items from the cache
313             String cacheName = getCacheName(type);
314             for (int i = 0; i < ids.length; i++) {
315                 if (ids[i] != null) {
316                     String key = ids[i].toString();
317                     getCacheProvider().remove(cacheName, key);
318                 }
319             }
320             // clear the search caches
321             getCacheProvider().clear(getSearchCacheName(type));
322         }
323     }

325     public <T> void saveSet(Set<T> entities) {
326         if (entities == null || entities.isEmpty()) {
327             System.out.println("WARN: Empty list of entities for saveSet, nothing to do...");
328         } else {
329             Class<?> type = checkEntitySet(entities);
330 
331             String operation = "saveSet";
332             beforeWrite(operation, type, null, entities.toArray());
333 
334             int changes = baseSaveSet(type, entities);
335 
336             afterWrite(operation, type, null, entities.toArray(), changes);
337 
338             // clear all saved items from the cache
339             String cacheName = getCacheName(type);
340             for (T t : entities) {
341                 Object id = baseGetIdValue(t);
342                 if (id != null) {
343                     String key = id.toString();
344                     getCacheProvider().remove(cacheName, key);
345                 }
346             }
347             // clear the search caches
348             getCacheProvider().clear(getSearchCacheName(type));
349         }
350     }

353         if (entities.size() > 0) {
354             Class<?> type = checkEntitySet(entities);
355             List<Object> ids = new ArrayList<Object>();
356             for (T t : entities) {

358                 if (id != null) {

361             }
362             deleteSet(type, ids.toArray(new Serializable[ids.size()]));
363         }

386     /***
387      * Validates the class type and the list of entities before performing
388      * a batch operation (throws IllegalArgumentException)
389      * 
390      * @param entities a Set of persistent entities, should all be of the same type
391      */
392     protected Class<?> checkEntitySet(Set<?> entities) {
393         Class<?> entityClass = null;
394         Iterator<?> it = entities.iterator();
395         while(it.hasNext()) {
396             Object entity = it.next();
397             if (entityClass == null) {
398                 entityClass = (Class<?>) findClass(entity);
399             }
400             if (! checkClass(entityClass).isInstance(entity)) {
401                 throw new IllegalArgumentException("Entity set item " +
402                         entity.toString() + " is not of type: " + entityClass +
403                         ", the type is: " + entity.getClass() +
404                 " (All items must be of consistent persistent type)");
405             }
406         }
407         return entityClass;
408     }
