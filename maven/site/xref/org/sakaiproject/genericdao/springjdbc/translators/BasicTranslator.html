25  /***
26   * A basic translator class to extend
27   * 
28   * @author Aaron Zeckoski (azeckoski@gmail.com)
29   */
30  public abstract class BasicTranslator implements DatabaseTranslator {
31  
32      /***
33       * Create a tablename from a given class
34       */
35      public static String makeTableNameFromClass(Class<?> pClass) {
36          String tableName = null;
37          // try to get from class annotations first
38          ClassFields<?> classFields = ReflectUtils.getInstance().analyzeClass(pClass);
39          if (classFields.getClassAnnotations().contains(PersistentTableName.class)) {
40              for (Annotation classAnnote : classFields.getClassAnnotations()) {
41                  if (PersistentTableName.class.equals(classAnnote.annotationType())) {
42                      tableName = ((PersistentTableName)classAnnote).value();
43                      break;
44                  }
45              }
46          }
47          if (tableName == null) {
48              String name = pClass.getSimpleName();
49              tableName = makeDBNameFromCamelCase(name);
50              if (tableName.startsWith("_") && tableName.length() > 1) {
51                  tableName = tableName.substring(1);
52              }
53          }
54          return tableName;
55      }
56  
57      /***
58       * Create a string that can be used for a DB name that is compatible with most databases (30 chars or less)
59       * @param camelCase a string in camelCase (e.g. someString, MyString)
60       * @return a string in DB style (e.g SOME_STRING, _MY_STRING)
61       */
62      public static String makeDBNameFromCamelCase(String camelCase) {
63          String dbCase;
64          if (camelCase.equals(camelCase.toUpperCase())) {
65              // name is already uppercase
66              dbCase = camelCase;
67          } else {
68              StringBuilder sb = new StringBuilder();
69              for (int i = 0; i < camelCase.length(); i++) {
70                  char c = camelCase.charAt(i);
71                  if (Character.isUpperCase(c)) {
72                      sb.append("_");
73                  }
74                  sb.append(c);
75              }
76              dbCase = sb.toString().toUpperCase();
77          }
78          return chopString(dbCase, 30);
79      }
80  
81      /***
82       * @param dbName a string in DB style (e.g SOME_STRING, MY_STRING, _MY_STRING)
83       * @return a string in camelCase (e.g. someString, myString, MyString)
84       */
85      public static String makeCamelCaseFromDBName(String dbName) {
86          String camelCase;
87          if (! dbName.equals(dbName.toUpperCase()) 
88                  && dbName.indexOf('_') == -1) {
89              // name is probably already fine
90              camelCase = dbName;
91          } else {
92              StringBuilder sb = new StringBuilder();
93              dbName = dbName.toLowerCase();
94              for (int i = 0; i < dbName.length(); i++) {
95                  char c = dbName.charAt(i);
96                  if (c == '_') {
97                      i++;
98                      try {
99                          c = Character.toUpperCase(dbName.charAt(i));
100                     } catch (IndexOutOfBoundsException e) {
101                         // end of string
102                         break;
103                     }
104                 }
105                 sb.append(c);
106             }
107             camelCase = sb.toString();
108         }
109         return camelCase;
110     }

112     /***
113      * Chop a string to the mexLength if it is longer and return the chopped string
114      * @param str
115      * @param maxLength
116      * @return the chopped string
117      */
118     public static String chopString(String str, int maxLength) {
119         if (str == null || "".equals(str)) { return ""; }
120         if (str.length() > maxLength) {
121             str = str.substring(0, 29);
122         }
123         return str;
124     }
