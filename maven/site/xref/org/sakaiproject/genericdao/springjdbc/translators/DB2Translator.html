23  public class DB2Translator extends BasicTranslator {

25     /* (non-Javadoc)
26      * @see org.sakaiproject.genericdao.api.translators.DatabaseTranslator#handlesDB()
27      */
28     public String handlesDB() {
29        return DBTYPE_DB2;
30     }

32     /* (non-Javadoc)
33      * @see org.sakaiproject.genericdao.api.translators.DatabaseTranslator#makeAutoIdQuery(java.lang.String)
34      */
35     public String makeAutoIdQuery(String tableName, String idColumnName) {
36        return "values identity_val_local()";
37     }

39     /* (non-Javadoc)
40      * @see org.sakaiproject.genericdao.api.translators.DatabaseTranslator#makeLimitQuery(java.lang.String, long, long)
41      */
42     public String makeLimitQuery(String sql, long start, long limit, String tableName) {
43        int startOfSelect = sql.toLowerCase().indexOf("select");
44  
45        StringBuilder pagingSelect = new StringBuilder( sql.length()+100 )
46                 .append( sql.substring(0, startOfSelect) ) //add the comment
47                 .append("select * from ( select ") //nest the main query in an outer select
48                 .append( getRowNumber(sql) ); //add the rownnumber bit into the outer query select list
49  
50        if ( hasDistinct(sql) ) {
51           pagingSelect.append(" row_.* from ( ") //add another (inner) nested select
52              .append( sql.substring(startOfSelect) ) //add the main query
53              .append(" ) as row_"); //close off the inner nested select
54        } else {
55           pagingSelect.append( sql.substring( startOfSelect + 6 ) ); //add the main query
56        }
57  
58        pagingSelect.append(" ) as temp_ where rownumber_ ");
59  
60        //add the restriction to the outer select
61        if (start > 0) {
62           pagingSelect.append("between ?+1 and ?");
63        } else {
64           pagingSelect.append("<= ?");
65        }
66        return pagingSelect.toString();
67     }
68  
69     /***
70      * true if this query is for a distinct result
71      */
72     private static boolean hasDistinct(String sql) {
73        return sql.toLowerCase().indexOf("select distinct")>=0;
74     }
75  
76     /***
77      * Render the <tt>rownumber() over ( .... ) as rownumber_,</tt>
78      * bit, that goes in the select list
79      */
80     private String getRowNumber(String sql) {
81        StringBuilder rownumber = new StringBuilder(50).append("rownumber() over(");
82  
83        int orderByIndex = sql.toLowerCase().indexOf("order by");
84  
85        if ( orderByIndex>0 && !hasDistinct(sql) ) {
86           rownumber.append( sql.substring(orderByIndex) );
87        }
88  
89        rownumber.append(") as rownumber_,");
90  
91        return rownumber.toString();
92     }
