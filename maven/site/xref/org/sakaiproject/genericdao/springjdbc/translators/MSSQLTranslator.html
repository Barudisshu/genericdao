15  package org.sakaiproject.genericdao.springjdbc.translators;

17  /***
18   * Microsoft SQL Server (MSSQL) translator
19   * 
20   * @author Aaron Zeckoski (azeckoski@gmail.com)
21   */
22  public class MSSQLTranslator extends BasicTranslator {
23  
24     public String handlesDB() {
25        return DBTYPE_MSSQL;
26     }
27  
28     public String makeAutoIdQuery(String tableName, String idColumnName) {
29        return "SELECT SCOPE_IDENTITY()";
30     }
31  
32     public String makeLimitQuery(String sql, long start, long limit, String tableName) {
33        // NOTE: no support for start/offset
34        return new StringBuilder( sql.length()+8 )
35           .append(sql)
36           .insert( getAfterSelectInsertPoint(sql), " top " + limit )
37           .toString();
38     }
39  
40     private int getAfterSelectInsertPoint(String sql) {
41        int selectIndex = sql.toLowerCase().indexOf( "select" );
42        final int selectDistinctIndex = sql.toLowerCase().indexOf( "select distinct" );
43        return selectIndex + ( selectDistinctIndex == selectIndex ? 15 : 6 );
44     }

46  }
