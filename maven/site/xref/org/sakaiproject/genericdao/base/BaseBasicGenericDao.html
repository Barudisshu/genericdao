66     public <T> long countBySearch(Class<T> type, Search search) {
67        checkClass(type);
68        if (search == null) {
69            throw new IllegalArgumentException("search cannot be null");
70        }
71        long count = 0;
72  
73        // check the cache first
74        boolean usedCache = false;
75        String searchCacheName = getSearchCacheName(type);
76        String cacheKey = "countBySearch::" + type.getName() + ":" + search.toString();
77        if (getCacheProvider().exists(searchCacheName, cacheKey)) {
78           Long lCount = (Long) getCacheProvider().get(searchCacheName, cacheKey);
79           if (lCount != null) {
80              count = lCount.longValue();
81              usedCache = true;
82           }
83        }
84  
85        if (! usedCache) {
86           count = baseCountBySearch(type, search);
87  
88           // cache the id results for the search
89           getCacheProvider().put(searchCacheName, cacheKey, Long.valueOf(count));
90        }
91        return count;
92     }

94     @SuppressWarnings("unchecked")
95     public <T> List<T> findBySearch(Class<T> type, Search search) {
96        checkClass(type);
97        if (search == null) {
98            throw new IllegalArgumentException("search cannot be null");
99        }
100       List<T> results = new ArrayList<T>();
101 
102       // check the cache first
103       boolean usedCache = false;
104       String cacheName = getCacheName(type);
105       String searchCacheName = getSearchCacheName(type);
106       String cacheKey = "findBySearch::" + type.getName() + ":" + search.toString();
107       if (getCacheProvider().exists(searchCacheName, cacheKey)) {
108          String[] resultIds = (String[]) getCacheProvider().get(searchCacheName, cacheKey);
109          if (resultIds != null) {
110             for (int i = 0; i < resultIds.length; i++) {
111                if (! getCacheProvider().exists(cacheName, resultIds[i])) {
112                   usedCache = false;
113                   break;
114                }
115                T entity = (T) getCacheProvider().get(cacheName, resultIds[i]);
116                results.add(entity);
117             }
118             usedCache = true;
119          }
120       }
121 
122       if (! usedCache) {
123          String operation = "findBySearch";
124          beforeRead(operation, type, null, search);
125 
126          results = baseFindBySearch(type, search);
127 
128          // run through the returned items for the interceptor and for caching
129          List<String> keys = new ArrayList<String>();
130          for (T entity : results) {
131             Object id = baseGetIdValue(entity);
132             // cache each returned item
133             String key = id.toString();
134             keys.add(key);
135             getCacheProvider().put(cacheName, key, entity);
136          }
137          // cache the id results for the search
138          String[] ids = keys.toArray(new String[keys.size()]);
139          getCacheProvider().put(searchCacheName, cacheKey, ids);
140          // call the after interceptor
141          afterRead(operation, type, ids, search, results.toArray(new Object[results.size()]));
142       }
143       return results;
144    }

146    @SuppressWarnings("unchecked")
147    public <T> T findOneBySearch(Class<T> type, Search search) {
148       checkClass(type);
149       if (search == null) {
150           throw new IllegalArgumentException("search cannot be null");
151       }
152       T entity = null;
153 
154       // check the cache first
155       boolean usedCache = false;
156       String cacheName = getCacheName(type);
157       String searchCacheName = getSearchCacheName(type);
158       String cacheKey = "findOneBySearch::" + type.getName() + ":" + search.toString();
159       if (getCacheProvider().exists(searchCacheName, cacheKey)) {
160          usedCache = true;
161          String id = (String) getCacheProvider().get(searchCacheName, cacheKey);
162          if (id != null) {
163             if (getCacheProvider().exists(cacheName, id)) {
164                entity = (T) getCacheProvider().get(cacheName, id);
165             }
166          }
167       }
168 
169       if (! usedCache) {
170          String operation = "findOneBySearch";
171          beforeRead(operation, type, null, search);
172 
173          search.setLimit(1); // only return 1 item
174 
175          entity = baseFindOneBySearch(type, search);
176 
177          String key = null;
178          if (entity != null) {
179             Serializable id = baseGetIdValue(entity);
180             afterRead(operation, type, new Serializable[] {id}, search, new Object[] {entity});
181 
182             if (id != null) {
183                // cache the entity
184                key = id.toString();
185                getCacheProvider().put(cacheName, key, entity);
186             }
187          }
188          // cache the search result
189          getCacheProvider().put(searchCacheName, cacheKey, key);
190       }
191       return entity;
192    }

265    /***
266     * @deprecated
267     */
268    @SuppressWarnings("unchecked")
269    public List findByProperties(Class entityClass, String[] objectProperties, Object[] values,
270          int[] comparisons, String[] sortProperties, int firstResult, int maxResults) {
271       if (objectProperties.length != values.length || values.length != comparisons.length) {
272          throw new IllegalArgumentException("All input arrays must be the same size");
273       }
274       Search search = new Search();
275       for (int i = 0; i < values.length; i++) {
276          search.addRestriction( new Restriction(objectProperties[i], values[i], comparisons[i]) );
277       }
278       if (sortProperties != null) {
279          for (int i = 0; i < sortProperties.length; i++) {
280             int location = sortProperties[i].indexOf(" ");
281             String property = sortProperties[i];
282             if (location > 0) {
283                property = sortProperties[i].substring(0, location);
284             }
285             Order order = null;
286             if (sortProperties[i].endsWith(ByPropsFinder.DESC)) {
287                order = new Order(property, false);
288             } else {
289                order = new Order(property, true);
290             }
291             search.addOrder( order );
292          }
293       }
294       search.setStart(firstResult);
295       search.setLimit(maxResults);
296       return findBySearch(entityClass, search);
297    }
