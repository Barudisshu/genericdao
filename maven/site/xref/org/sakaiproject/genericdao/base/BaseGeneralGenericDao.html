61     public <T> int countAll(Class<T> type) {
62        checkClass(type);
63        int count = 0;
64  
65        // check the cache first
66        boolean usedCache = false;
67        String searchCacheName = getSearchCacheName(type);
68        String cacheKey = "countAll::" + type.getName();
69        if (getCacheProvider().exists(searchCacheName, cacheKey)) {
70           Integer iCount = (Integer) getCacheProvider().get(searchCacheName, cacheKey);
71           if (iCount != null) {
72              count = iCount.intValue();
73              usedCache = true;
74           }
75        }
76  
77        if (! usedCache) {
78           count = baseCountAll(type);
79  
80           // cache the id results for the search
81           getCacheProvider().put(searchCacheName, cacheKey, Integer.valueOf(count));
82        }
83        return count;
84     }

86     public <T> void deleteSet(Class<T> type, Serializable[] ids) {
87        checkClass(type);
88        if (ids.length > 0) {
89           String operation = "deleteSet";
90           beforeWrite(operation, type, ids, null);
91     
92           int changes = baseDeleteSet(type, ids);
93     
94           afterWrite(operation, type, ids, null, changes);
95     
96           // clear all removed items from the cache
97           String cacheName = getCacheName(type);
98           for (int i = 0; i < ids.length; i++) {
99              if (ids[i] != null) {
100                String key = ids[i].toString();
101                getCacheProvider().remove(cacheName, key);
102             }
103          }
104          // clear the search caches
105          getCacheProvider().clear(getSearchCacheName(type));
106       }
107    }

109    public <T> void saveSet(Set<T> entities) {
110       if (entities == null || entities.isEmpty()) {
111          System.out.println("WARN: Empty list of entities for saveSet, nothing to do...");
112       } else {
113          Class<?> type = checkEntitySet(entities);
114    
115          String operation = "saveSet";
116          beforeWrite(operation, type, null, entities.toArray());
117    
118          int changes = baseSaveSet(type, entities);
119    
120          afterWrite(operation, type, null, entities.toArray(), changes);
121    
122          // clear all saved items from the cache
123          String cacheName = getCacheName(type);
124          for (T t : entities) {
125             Object id = baseGetIdValue(t);
126             if (id != null) {
127                String key = id.toString();
128                getCacheProvider().remove(cacheName, key);
129             }
130          }
131          // clear the search caches
132          getCacheProvider().clear(getSearchCacheName(type));
133       }
134    }

136    public <T> void deleteSet(Set<T> entities) {
137       if (entities.size() > 0) {
138          Class<?> type = checkEntitySet(entities);
139          List<Object> ids = new ArrayList<Object>();
140          for (T t : entities) {
141             Object id = baseGetIdValue(t);
142             if (id != null) {
143                ids.add(id);
144             }
145          }
146          deleteSet(type, ids.toArray(new Serializable[ids.size()]));
147       }
148    }

170    /***
171     * Validates the class type and the list of entities before performing
172     * a batch operation (throws IllegalArgumentException)
173     * 
174     * @param entities a Set of persistent entities, should all be of the same type
175     */
176    protected Class<?> checkEntitySet(Set<?> entities) {
177       Class<?> entityClass = null;
178       Iterator<?> it = entities.iterator();
179       while(it.hasNext()) {
180          Object entity = it.next();
181          if (entityClass == null) {
182             entityClass = (Class<?>) findClass(entity);
183          }
184          if (! checkClass(entityClass).isInstance(entity)) {
185             throw new IllegalArgumentException("Entity set item " +
186                   entity.toString() + " is not of type: " + entityClass +
187                   ", the type is: " + entity.getClass() +
188             " (All items must be of consistent persistent type)");
189          }
190       }
191       return entityClass;
192    }
