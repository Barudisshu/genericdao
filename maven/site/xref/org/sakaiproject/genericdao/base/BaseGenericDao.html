69     /***
70      * Get the id value from a persistent object,
71      * this expects that the object is persistent so it should already be checked<br/>
72      * Override this if desired
73      * @param entity any persistent object
74      * @return the id value or null if it is unset
75      */
76     protected Serializable baseGetIdValue(Object entity) {
77        Class<?> type = findClass(entity);
78        String idProp = getIdProperty(type);
79        Object idValue = null;
80        if (idProp == null) {
81           throw new IllegalArgumentException("Could not find id property for this class type: " + type);
82        } else {
83           idValue = ReflectUtils.getInstance().getFieldValue(entity, idProp);
84        }
85        Serializable serialId = null;
86        if (idValue != null) {
87           if (! (idValue instanceof Serializable)) {
88              serialId = idValue.toString();
89           }
90        }
91        return serialId;
92     }

94     /***
95      * Find the class type of a persistent object,
96      * needed in the case that we are working with something that wraps it's objects<br/>
97      * Override this if desired
98      * @param entity a persistent entity
99      * @return the persistent class type OR null if it cannot be found
100     */
101    protected Class<?> findClass(Object entity) {
102       if (entity == null) {
103          throw new IllegalArgumentException("Cannot find class type of null entity object");
104       }
105       Class<?> type = entity.getClass();
106       return type;
107    }

111    private List<Class<?>> classes;
112    /***
113     * This does a nice bit of exception handling for us and verifies that
114     * this class is valid to perform a DAO operation with
115     * @param type class type of the persistent object to check
116     * @return A valid entityClass type resolved to be the same as
117     * the ones usable by this DAO
118     */
119    protected Class<?> checkClass(Class<?> type) {
120       if (type == null) {
121          throw new NullPointerException("type cannot be null");
122       }
123 
124       if (classes == null) {
125          throw new NullPointerException("persistent classes must be set");
126       }
127 
128       for (Iterator<Class<?>> i = classes.iterator(); i.hasNext();) {
129          Class<?> concrete = (Class<?>) i.next();
130          if (concrete.isAssignableFrom(type)) {
131             return concrete;
132          }
133       }
134       throw new IllegalArgumentException("Could not resolve this class " + 
135             type + " as part of the set of persistent objects: " +
136             classes.toString());
137    }

188    private Map<Class<?>, ReadInterceptor> readInterceptors = new ConcurrentHashMap<Class<?>, ReadInterceptor>();
189    private Map<Class<?>, WriteInterceptor> writeInterceptors = new ConcurrentHashMap<Class<?>, WriteInterceptor>();
190    /***
191     * Adds the provided interceptor to the current set of interceptors
192     * @param interceptor
193     */
194    public void addInterceptor(DaoOperationInterceptor interceptor) {
195       if (interceptor != null) {
196          Class<?> type = interceptor.interceptType();
197          if (type != null) {
198             if (ReadInterceptor.class.isAssignableFrom(interceptor.getClass())) {
199                readInterceptors.put(type, (ReadInterceptor) interceptor);
200             }
201             if (WriteInterceptor.class.isAssignableFrom(interceptor.getClass())) {
202                writeInterceptors.put(type, (WriteInterceptor) interceptor);
203             }
204          }
205       }
206    }
207    /***
208     * Removes the provided interceptor from the current set of interceptors
209     * @param interceptor
210     */
211    public void removeInterceptor(DaoOperationInterceptor interceptor) {
212       if (interceptor != null) {
213          Class<?> type = interceptor.interceptType();
214          if (type != null) {
215             if (ReadInterceptor.class.isAssignableFrom(interceptor.getClass())) {
216                readInterceptors.remove(type);
217             }
218             if (WriteInterceptor.class.isAssignableFrom(interceptor.getClass())) {
219                writeInterceptors.remove(type);
220             }
221          }
222       }
223    }
224    /***
225     * Makes the given interceptor the only active interceptor,
226     * removes all others and adds only this one
227     * @param interceptor
228     */
229    public void setInterceptor(DaoOperationInterceptor interceptor) {
230       if (interceptor != null) {
231          readInterceptors.clear();
232          writeInterceptors.clear();
233          addInterceptor(interceptor);
234       }
235    }

290    /* (non-Javadoc)
291     * @see org.sakaiproject.genericdao.api.GenericDao#findById(java.lang.Class, java.io.Serializable)
292     */
293    @SuppressWarnings("unchecked")
294    public <T> T findById(Class<T> type, Serializable id) {
295       checkClass(type);
296       if (id == null) {
297          throw new IllegalArgumentException("id must be set to find persistent object");
298       }
299 
300       T entity = null;
301       // check cache first
302       String key = id.toString();
303       String cacheName = getCacheName(type);
304       if (getCacheProvider().exists(cacheName, key)) {
305          entity = (T) getCacheProvider().get(cacheName, key);
306       } else {
307          // not in cache so go to the DB
308    
309          // before interceptor
310          String operation = "findById";
311          beforeRead(operation, type, new Serializable[] {id}, null);
312    
313          entity = baseFindById(type, id);
314    
315          // now put the item in the cache
316          getCacheProvider().put(cacheName, key, entity);
317    
318          // after interceptor
319          afterRead(operation, type, new Serializable[] {id}, null, new Object[] {entity});
320       }
321       return entity;
322    }

327    public void create(Object object) {
328       Class<?> type = findClass(object);
329       checkClass(type);
330 
331       String operation = "create";
332       beforeWrite(operation, type, null, new Object[] {object});
333 
334       Serializable idValue = baseCreate(type, object);
335 
336       // clear the search caches
337       getCacheProvider().clear(getSearchCacheName(type));
338 
339       afterWrite(operation, type, new Serializable[] {idValue}, new Object[] {object}, 1);
340    }

342    /* (non-Javadoc)
343     * @see org.sakaiproject.genericdao.api.modifiers.BasicModifier#update(java.lang.Object)
344     */
345    public void update(Object object) {
346       Class<?> type = findClass(object);
347       checkClass(type);
348 
349       Serializable idValue = baseGetIdValue(object);
350       if (idValue == null) {
351          throw new IllegalArgumentException("Could not get an id value from the supplied object, cannot update without an id: " + object);
352       }
353 
354       String operation = "update";
355       beforeWrite(operation, type, new Serializable[] {idValue}, new Object[] {object});
356 
357       baseUpdate(type, idValue, object);
358 
359       // clear the search caches
360       getCacheProvider().clear(getSearchCacheName(type));
361       // clear the cache entry since this was updated
362       String key = idValue.toString();
363       String cacheName = getCacheName(type);
364       getCacheProvider().remove(cacheName, key);
365 
366       afterWrite(operation, type, new Serializable[] {idValue}, new Object[] {object}, 1);
367    }

390    /* (non-Javadoc)
391     * @see org.sakaiproject.genericdao.api.modifiers.BasicModifier#delete(java.lang.Class, java.io.Serializable)
392     */
393    public <T> boolean delete(Class<T> type, Serializable id) {
394       checkClass(type);
395 
396       String operation = "delete";
397       beforeWrite(operation, type, new Serializable[] {id}, null);
398 
399       boolean removed = baseDelete(type, id);
400 
401       if (removed) {
402          // clear the search caches
403          getCacheProvider().clear(getSearchCacheName(type));
404          // clear this from the cache
405          String key = id.toString();
406          String cacheName = getCacheName(type);
407          getCacheProvider().remove(cacheName, key);
408          
409          afterWrite(operation, type, new Serializable[] {id}, null, 1);
410       }
411       return removed;
412    }
