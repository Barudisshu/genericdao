15  package org.sakaiproject.genericdao.test;

17  import java.util.HashMap;
18  import java.util.List;
19  import java.util.Map;
20  import java.util.Vector;
21  import java.util.concurrent.ConcurrentHashMap;
22  
23  import org.sakaiproject.genericdao.api.caching.CacheKeyNotFoundException;
24  import org.sakaiproject.genericdao.api.caching.CacheProvider;
25  
26  /***
27   * This is a very simple in memory map cache,
28   * it is meant for use in testing and as an example only
29   * 
30   * @author Aaron Zeckoski (azeckoski@gmail.com)
31   */
32  public class BasicMapCacheProvider implements CacheProvider {
33  
34     private Map<String, Map<String, Object>> cache = new ConcurrentHashMap<String, Map<String,Object>>();
35  
36     protected List<String> actionRecord = new Vector<String>();
37     public List<String> getActionRecord() {
38        return actionRecord;
39     }
40     public String getLastAction() {
41        String action = null;
42        if (actionRecord.size() > 0) {
43           action = actionRecord.get(actionRecord.size()-1);
44        }
45        return action;
46     }
47     public int size(String cacheName) {
48        checkCacheName(cacheName);
49        return cache.get(cacheName).size();      
50     }
51     public void reset() {
52        for (Map<String, Object> m : cache.values()) {
53           m.clear();
54        }
55        actionRecord.clear();
56     }
57  
58     public void clear(String cacheName) {
59        checkCacheName(cacheName);
60        cache.get(cacheName).clear();
61        actionRecord.add("clear:" + cacheName);
62     }
63  
64     public void createCache(String cacheName) {
65        if (cacheName == null) {
66           throw new IllegalArgumentException("cacheName cannot be null");
67        }
68        if (! cache.containsKey(cache)) {
69           Map<String, Object> iCache = new HashMap<String, Object>();
70           cache.put(cacheName, iCache);
71           actionRecord.add("create:" + cacheName + ":new");
72        } else {
73           cache.get(cacheName).clear();
74           actionRecord.add("create:" + cacheName + ":clear");
75        }
76     }
77  
78     public boolean exists(String cacheName, String key) {
79        checkCacheName(cacheName);
80        boolean exists = cache.get(cacheName).containsKey(key);
81        actionRecord.add("exists: " + cacheName + ":" + key + ":exists=" + exists);
82        return exists;
83     }
84  
85     public Object get(String cacheName, String key) {
86        checkCacheName(cacheName);
87        if (! cache.get(cacheName).containsKey(key)) {
88           throw new CacheKeyNotFoundException("Key ("+key+") does not exist in cache ("+cacheName+")", cacheName, key);
89        }
90        actionRecord.add("get: " + cacheName + ":" + key);
91        return cache.get(cacheName).get(key);
92     }
93  
94     public void put(String cacheName, String key, Object value) {
95        checkCacheName(cacheName);
96        if (key == null) {
97           throw new IllegalArgumentException("key cannot be null");
98        }
99        actionRecord.add("put: " + cacheName + ":" + key + ":value=" + value);
100       cache.get(cacheName).put(key, value);
101    }
102 
103    public boolean remove(String cacheName, String key) {
104       checkCacheName(cacheName);
105       boolean exists = false;
106       if (cache.get(cacheName).containsKey(key)) {
107          exists = true;
108          cache.get(cacheName).remove(key);
109       }
110       actionRecord.add("remove: " + cacheName + ":" + key + ":removed=" + exists);
111       return exists;
112    }
113 
114    private void checkCacheName(String cacheName) {
115       if (cacheName == null) {
116          throw new IllegalArgumentException("cacheName cannot be null");
117       }
118       if (! cache.containsKey(cacheName)) {
119          throw new IllegalArgumentException("Invalid cache name: " + cacheName);
120       }
121    }
122 }
