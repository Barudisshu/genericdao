17  import java.io.Serializable;
18  import java.util.ArrayList;
19  import java.util.List;
20  import java.util.Vector;

22  import org.sakaiproject.genericdao.api.interceptors.ReadInterceptor;
23  import org.sakaiproject.genericdao.api.interceptors.WriteInterceptor;
24  import org.sakaiproject.genericdao.api.search.Search;

27  /***
28   * An example DAO interceptor, mostly for testing purposes but is an ok example of what one might look like,
29   * this intercepts DAO operations on {@link GenericTestObject}
30   * 
31   * @author Aaron Zeckoski (azeckoski@gmail.com)
32   */
33  public class BasicDataInterceptor implements ReadInterceptor, WriteInterceptor {
34     public String lastIntercept;
35     public String lastOperation;
36     public Serializable[] lastIds;
37     public Search lastSearch;
38     public Object[] lastEntities;
39     public int lastChanges;
40  
41     protected List<Intercept> intercepts = new Vector<Intercept>();
42     /***
43      * @return the list of all intercepts made by this interceptor
44      */
45     public List<Intercept> getIntercepts() {
46        return new ArrayList<Intercept>(intercepts);
47     }
48     /***
49      * @return the most recent intercept or null if there are none
50      */
51     public Intercept getLastIntercept() {
52        return intercepts.size() > 0 ? intercepts.get(intercepts.size()-1) : null;
53     }
54     /***
55      * reset the list of intercepts
56      */
57     public void reset() {
58        intercepts.clear();
59        lastIntercept = null;
60        lastOperation = null;
61        lastIds = null;
62        lastSearch = null;
63        lastEntities = null;
64        lastChanges = 0;
65     }
66  
67     private void setLasts(String method, String operation, Serializable[] ids, Search search,
68           Object[] entities, int changes) {
69        this.lastIntercept = method;
70        this.lastOperation = operation;
71        this.lastIds = ids;
72        this.lastSearch = search;
73        this.lastEntities = entities;
74        this.lastChanges = changes;
75        intercepts.add( new Intercept(method, operation, ids, search, entities, changes) );
76     }
77  
78     public Class<?> interceptType() {
79        return GenericTestObject.class;
80     }
81  
82     public void afterRead(String operation, Serializable[] ids, Search search, Object[] entities) {
83        setLasts("afterRead", operation, ids, search, entities, 0);
84     }
85  
86     public void beforeRead(String operation, Serializable[] ids, Search search) {
87        setLasts("beforeRead", operation, ids, search, null, 0);
88     }
89  
90     public void afterWrite(String operation, Serializable[] ids, Object[] entities, int changes) {
91        setLasts("afterWrite", operation, ids, null, entities, changes);
92     }
93  
94     public void beforeWrite(String operation, Serializable[] ids, Object[] entities) {
95        setLasts("beforeWrite", operation, ids, null, entities, 0);
96     }
97  
98     public class Intercept {
99        public String intercept;
100       public String operation;
101       public Serializable[] ids;
102       public Search search;
103       public Object[] entities;
104       public int changes;
105       public Intercept(String intercept, String operation, Serializable[] ids, Search search,
106             Object[] entities, int changes) {
107          this.intercept = intercept;
108          this.operation = operation;
109          this.ids = ids;
110          this.search = search;
111          this.entities = entities;
112          this.changes = changes;
113       }
114       @Override
115       public String toString() {
116          return "intercept=" + intercept + ",operation=" + operation + ",ids=" + ids + ",search=" + search + ",changes=" + changes;
117       }
118    }
