26  /***
27   * Mapper for the Generic Test Parent
28   * 
29   * @author Aaron Zeckoski (azeckoski@gmail.com)
30   */
31  public class GTPDataMapper implements DataMapper, EntityColumnMapper {
32  
33     public String getIdPropertyName() {
34        return "uid";
35     }
36  
37     public Class<?> getPersistentType() {
38        return GenericTestParentObject.class;
39     }
40  
41     public NamesRecord getPropertyToColumnNamesMapping() {
42        NamesRecord nr = new NamesRecord();
43        nr.setNameMapping("uid", "GTP_ID");
44        nr.setNameMapping("title", "GTP_TITLE");
45        nr.setNameMapping("gto", "GTP_GTO");
46        return nr;
47     }
48  
49     public String getTableName() {
50        return "SUPER_GENERIC_PARENT";
51     }
52  
53     public String generateDDL(String databaseTypeConstant) {
54        StringBuilder sql = new StringBuilder();
55        if (DBTYPE_HSQLDB.equals(databaseTypeConstant)) {
56           sql.append("CREATE TABLE {TABLENAME} ( " +
57                 "{ID}               BIGINT NOT NULL IDENTITY PRIMARY KEY," +
58                 "GTP_TITLE          VARCHAR(255) NOT NULL," +
59                 "GTP_GTO            BIGINT," +
60                 "FOREIGN KEY (GTP_GTO) REFERENCES {TABLENAME:GenericTestObject}({ID:GenericTestObject})" +
61           ");\n");
62        } else if (DBTYPE_MYSQL.equals(databaseTypeConstant)) {
63           sql.append("CREATE TABLE {TABLENAME} ( " +
64                 "{ID}               bigint(20) AUTO_INCREMENT NOT NULL PRIMARY KEY," +
65                 "GTP_TITLE          varchar(255) NOT NULL," +
66                 "GTP_GTO            bigint(20)," +
67                 "FOREIGN KEY (GTP_GTO) REFERENCES {TABLENAME:GenericTestObject}({ID:GenericTestObject})" +
68           ");\n");
69        } else {
70           throw new UnsupportedOperationException("No support for the database type: " + databaseTypeConstant);
71        }
72        return sql.toString();      
73     }
74  
75     public Object mapColumnsToObject(Map<String, Object> columnsData) {
76        GenericTestParentObject gtpo = new GenericTestParentObject();
77        for (String key : columnsData.keySet()) {
78           Object value = columnsData.get(key);
79           if ("GTP_ID".equals(key)) {
80              gtpo.setUid((Long)value);
81           } else if ("GTP_TITLE".equals(key)) {
82              gtpo.setTitle((String)value);
83           } else if ("GTP_GTO".equals(key)) {
84              GenericTestObject gto = new GenericTestObject();
85              gto.setId((Long)value);
86              gtpo.setGto(gto);
87           }
88        }
89        return null;
90     }
91  
92     public Map<String, Object> mapObjectToColumns(Object persistentObject) {
93        GenericTestParentObject gtpo = (GenericTestParentObject) persistentObject;
94        Map<String, Object> map = new HashMap<String, Object>();
95        map.put("GTP_ID", gtpo.getUid());
96        map.put("GTP_TITLE", gtpo.getTitle());
97        map.put("GTP_GTO", gtpo.getGto() == null ? null : gtpo.getGto().getId());
98        return map;
99     }
