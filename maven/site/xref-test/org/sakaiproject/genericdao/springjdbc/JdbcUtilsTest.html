177     */
178    public void testHandleTypeReplacements() {
179       String ddl = null;
180 
181       ddl = "create {TABLENAME} ({ID} varchar, other varchar)";
182       ddl = dao.handleTypeReplacements(GenericTestObject.class, ddl);
183       assertNotNull(ddl);
184       assertEquals("create MY_GENERIC_TEST_OBJECT (GTO_ID varchar, other varchar)", ddl);
185 
186       ddl = "create {TABLENAME} ({ID} varchar, other varchar)";
187       ddl = dao.handleTypeReplacements(GenericTestParentObject.class, ddl);
188       assertNotNull(ddl);
189       assertEquals("create GENERIC_TEST_PARENT_OBJECT (UID varchar, other varchar)", ddl);
190 
191       ddl = "create {TABLENAME} ({ID} varchar, other varchar, " +
192       "FOREIGN KEY (other) REFERENCES {TABLENAME:"+GenericTestObject.class.getName()+"}({ID:"+GenericTestObject.class.getName()+"}) )";
193       ddl = dao.handleTypeReplacements(GenericTestParentObject.class, ddl);
194       assertNotNull(ddl);
195       assertEquals("create GENERIC_TEST_PARENT_OBJECT (UID varchar, other varchar, " +
196             "FOREIGN KEY (other) REFERENCES MY_GENERIC_TEST_OBJECT(GTO_ID) )", ddl);
197 
198       ddl = "create {TABLENAME} ({ID} varchar, other varchar, " +
199       "FOREIGN KEY (other) REFERENCES {TABLENAME:GenericTestObject}({ID:GenericTestObject}) )";
200       ddl = dao.handleTypeReplacements(GenericTestParentObject.class, ddl);
201       assertNotNull(ddl);
202       assertEquals("create GENERIC_TEST_PARENT_OBJECT (UID varchar, other varchar, " +
203             "FOREIGN KEY (other) REFERENCES MY_GENERIC_TEST_OBJECT(GTO_ID) )", ddl);
204 
205       ddl = "create {TABLENAME} ({ID} varchar, {COLUMNNAME:title} varchar)";
206       ddl = dao.handleTypeReplacements(GenericTestObject.class, ddl);
207       assertNotNull(ddl);
208       assertEquals("create MY_GENERIC_TEST_OBJECT (GTO_ID varchar, GTO_TITLE varchar)", ddl);
209 
210    }
